class LoRALinear(nn.Module):
    """Standard linear layer + low‑rank adaptation (LoRA)."""
    def __init__(self, in_features, out_features, rank=4):
        super().__init__()
        self.weight = nn.Parameter(torch.randn(out_features, in_features) * 0.01)
        self.bias   = nn.Parameter(torch.zeros(out_features))
        # LoRA matrices
        self.lora_A = nn.Parameter(torch.randn(rank, in_features) * 0.01)
        self.lora_B = nn.Parameter(torch.zeros(out_features, rank))
        self.scale = 1.0 / rank          # usual LoRA scaling

    def forward(self, x):
        # W + (B @ A)   (note: x @ W.T  +  x @ A.T @ B.T)
        return (
            x @ self.weight.t()
            + self.bias
            + (x @ self.lora_A.t() @ self.lora_B.t()) * self.scale
        )

# --------------------------------------------------------------
# 4️⃣  Tiny network (one LoRA layer + a classifier head)
# --------------------------------------------------------------

    def __init__(self):

        self.l1 = LoRALinear(20, 50, rank=4)   # <-- LoRA lives here

        self.l2 = nn.Linear(50, 1)             # logistic regression head

    def forward(self, x):
        x = self.act(self.l1(x))
        return self.l2(x).squeeze(-1)
