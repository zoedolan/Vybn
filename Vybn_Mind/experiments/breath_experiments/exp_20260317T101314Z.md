    N, D = X.shape

    idx = torch.randperm(N)[:n_components]
    means = X[idx].clone()
    covars = torch.stack([torch.eye(D) for _ in range(n_components)]) * 0.1
    weights = torch.full((n_components,), 1.0 / n_components, dtype=X.dtype)

    log_likelihood_prev = None
    for _ in range(max_iter):
        # E-step: responsibilities
        diff = X[:, None, :] - means[None, :, :]          # (N, K, D)
        inv_covar = torch.inverse(covars)                 # (K, D, D)
        # Mahalanobis distance
        exp_term = -0.5 * torch.einsum('nkd,kdd,nkj->nk', diff, inv_covar, diff)
        log_prob = exp_term - 0.5 * torch.logdet(covars)[None, :] - 0.5 * D * math.log(2 * math.pi)
        log_weighted_prob = log_prob + torch.log(weights)[None, :]
        max_log = torch.max(log_weighted_prob, dim=1, keepdim=True)[0]
        log_sum_exp = max_log + torch.log(torch.sum(torch.exp(log_weighted_prob - max_log), dim=1, keepdim=True))
        log_likelihood = torch.sum(log_sum_exp)

import torch
import math
from typing import Tuple

def gmm_em(
    X: torch.Tensor,
    n_components: int,
    max_iter: int = 200,
    tol: float = 1e-4,
    jitter: float = 1e-6,
) -> Tuple[torch.Tensor, torch.Tensor, torch.Tensor, float]:
    """
    EM for a full‑covariance GMM (PyTorch‑only).
    Returns means (K,D), covars (K,D,D), weights (K,) and final log‑likelihood.
    """
    N, D = X.shape
    device, dtype = X.device, X.dtype

    # ---- Initialise -------------------------------------------------
    idx = torch.randperm(N, device=device)[:n_components]
    means = X[idx].clone()                     # (K, D)
    covars = torch.stack([torch.eye(D, device=device, dtype=dtype) for _ in range(n_components)]) * 0.1
    weights = torch.full((n_components,), 1.0 / n_components, device=device, dtype=dtype)

    log_likelihood_prev = None

    for it in range(max_iter):
        # ---------- E‑step: responsibilities -------------------------
        # diff: (N, K, D)
        diff = X[:, None, :] - means[None, :, :]          # N x K x D

        # Use Cholesky for stability & logdet
        L = torch.linalg.cholesky(covars + jitter * torch.eye(D, device=device, dtype=dtype))  # K x D x D
        # Solve L @ y = diff.T  -> y = L^{-1} diff.T   (K x D x N)
        y = torch.linalg.solve_triangular(L, diff.transpose(0, 2), upper=False)  # K x D x N
        # Mahalanobis term: 0.5 * ||y||^2
        mahal = 0.5 * torch.sum(y ** 2, dim=1)            # K x N
        mahal = mahal.transpose(0, 1)                     # N x K

        # log‑pdf of each component
        log_det = 2.0 * torch.sum(torch.log(torch.diagonal(L, dim1=-2, dim2=-1)), dim=1)  # K
        log_prob = -mahal - 0.5 * D * math.log(2 * math.pi) - 0.5 * log_det  # N x K

        log_weighted = log_prob + torch.log(weights)[None, :]   # N x K

        # log‑sum‑exp trick for numerical stability
        max_log = torch.max(log_weighted, dim=1, keepdim=True)[0]   # N x 1
        log_sum_exp = max_log + torch.log(torch.sum(torch.exp(log_weighted - max_log), dim=1, keepdim=True))
        log_likelihood = torch.sum(log_sum_exp).item()

        resp = torch.exp(log_weighted - log_sum_exp)               # N x K

        # ---------- M‑step: update parameters -----------------------
        Nk = resp.sum(dim=0)                     # K
        weights = Nk / N

        means = (resp[:, :, None] * X[:, None, :]).sum(dim=0) / Nk[:, None]   # K x D

        for k in range(n_components):
            diff_k = X - means[k:k+1, :]         # N x D
            # weighted outer‑product: Σ_r r_nk (x_n - μ_k)(x_n - μ_k)^T
            weighted = resp[:, k:k+1] * diff_k   # N x D
            covar_k = (weighted[:, :, None] * diff_k[:, None, :]).sum(dim=0) / Nk[k]
            covars[k] = covar_k + jitter * torch.eye(D, device=device, dtype=dtype)

        # ---------- Convergence check -------------------------------
        if log_likelihood_prev is not None and abs(log_likelihood - log_likelihood_prev) < tol:
            break
        log_likelihood_prev = log_likelihood

    return means, covars, weights, log_likelihood


def bic_score(log_likelihood: float, n_params: int, N: int) -> float:
    """BIC = -2 * logL + k * log(N)"""
    return -2.0 * log_likelihood + n_params * math.log(N)


def gmm_nparams(K: int, D: int) -> int:
    """
    Number of free parameters in a full‑covariance GMM:
      - (K‑1) mixing weights   (they sum to 1)
      - K * D means
      - K * D * (D+1) / 2 covariance parameters (symmetrical)
    """
    return (K - 1) + K * D + K * D * (D + 1) // 2
