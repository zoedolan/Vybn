```python
import torch
import torch.nn as nn
import torch.nn.functional as F
import numpy as np

# Synthetic binary classification data
torch.manual_seed(0)
X = torch.randn(500, 20)
y = (X[:, 0] + X[:, 1] > 0).long()

# Simple model: linear layer + LoRA adapter (low-rank)
class LoRALinear(nn.Module):
    def __init__(self, in_features, out_features, rank=4):
        super().__init__()
        self.linear = nn.Linear(in_features, out_features, bias=False)
        self.lora_A = nn.Parameter(torch.randn(rank, in_features) * 0.01)
        self.lora_B = nn.Parameter(torch.randn(out_features, rank) * 0.01)
        self.rank = rank

    def forward(self, x):
        # W + BA
        return self.linear(x) + (x @ self.lora_A.T @ self.lora_B.T)

model = LoRALinear(20, 2)
optimizer = torch.optim.SGD(model.parameters(), lr=0.1)
criterion = nn.CrossEntropyLoss()

# Pretrain for a few epochs
for epoch in range(5):
    model.train()
    optimizer.zero_grad()
    logits = model(X)
    loss = criterion(logits, y)
    loss.backward()
    optimizer.step()

# Evaluate linear probe accuracy before hiatus
model.eval()
with torch.no_grad():
    acc_before = (model(X).argmax(dim=1) == y).float().mean().item()

# Hiatus: apply a tiny random perturbation to LoRA parameters
epsilon = 0.005  # perturbation magnitude
with torch.no_grad():
    model.lora_A += epsilon * torch.randn_like(model.lora_A)
    model.lora_B += epsilon * torch.randn_like(model.lora_B)

# Measure the drift (Frobenius norm of the perturbation)
drift = torch.norm(model.lora_A - (model.lora_A - epsilon * torch.randn_like(model.lora_A))).item() + \
        torch.norm(model.lora_B - (model.lora_B - epsilon * torch.randn_like(model.lora_B))).item()
# Actually compute the applied perturbation norm
perturb_norm = torch.norm(model.lora_A - (model.lora_A - epsilon * torch.randn_like(model.lora_A))).item() + \
               torch.norm(model.lora_B - (model.lora_B - epsilon * torch.randn_like(model.lora_B))).item()
# Simpler: directly compute norm of the added noise
with torch.no_grad():
    noise_A = epsilon * torch.randn_like(model.lora_A)
    noise_B = epsilon * torch.randn_like(model.lora_B)
    perturb_norm = torch.norm(noise_A).item() + torch.norm(noise_B).item()

print(f"Perturbation norm (drift): {perturb_norm:.6f}")

# Evaluate after hiatus
model.eval()
with torch.no_grad():
    acc_after = (model(X).argmax(dim=1) == y).float().mean().item()

print(f"Accuracy before hiatus: {acc_before:.4f}")
print(f"Accuracy after hiatus:  {acc_after:.4f}")

# Decision rule: if drift < epsilon_threshold (here 0.01) we expect no drop
epsilon_threshold = 0.01
if perturb_norm < epsilon_threshold:
    print("Drift below threshold → performance preserved (as expected).")
else:
    print("Drift above threshold → performance may degrade.")
```

### What you would need to know to guarantee *no* drop

The decision rule is based on the inequality  

\[
\|\Delta A\|_F + \|\Delta B\|_F < \varepsilon_{\text{threshold}} .
\]

If you want to be certain that the hiatus will not hurt performance, you must choose the perturbation magnitude `ε` such that  

\[
\varepsilon\big(\sqrt{\text{numel}(A)}+\sqrt{\text{numel}(B)}\big) < \varepsilon_{\text{threshold}} .
\]

With `numel(A)=80`, `numel(B)=8` and `ε_threshold=0.01`, this gives  

\[
\varepsilon < \frac{0.01}{\sqrt{80}+\sqrt{8}} \approx \frac{0.01}{8.944+2.828}\approx 0.00086 .
\]

So a perturbation on the order of **≤ 8·10⁻⁴** would keep the drift below the threshold and, in this simple setting, preserve accuracy.  

If you need to tolerate larger `ε`, you would either have to increase the threshold (accept a larger allowed drift) or improve the robustness of the model (e.g., by training with weight‑noise regularization, using a larger LoRA rank, or adding more training epochs so that the classifier relies less on the fragile LoRA component).
