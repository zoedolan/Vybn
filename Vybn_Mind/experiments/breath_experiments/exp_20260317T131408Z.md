## Proposal
PROBE
Test whether steps with low prediction error (low MSE) of the next weight update from recent history correspond to higher mutual information between successive updates and to smoother loss decline. We'll train a tiny network, record weight‑update vectors, compute both the MSE of a linear predictor and an MI estimate, and examine their relationship across training.

```python
import numpy as np
import torch
import torch.nn as nn
import torch.optim as optim
from sklearn.metrics import mutual_info_regression

# Simple model and synthetic data
torch.manual_seed(0)
model = nn.Sequential(nn.Linear(20, 50), nn.ReLU(), nn.Linear(50, 1))
optimizer = optim.SGD(model.parameters(), lr=0.01)
criterion = nn.MSELoss()

X = torch.randn(500, 20)
y = torch.randn(500, 1)

# Storage
updates = []          # list of flattened weight-update vectors per step
losses = []           # training loss per step
pred_mses = []        # MSE of predicting next update from previous 3 updates
mis = []              # Mutual information estimate between past and next update

window = 3  # how many past updates to use for prediction

for step in range(200):
    model.zero_grad()
    out = model(X)
    loss = criterion(out, y)
    loss.backward()
    optimizer.step()
    
    # Flatten weight update (theta_t - theta_{t-1})
    flat_update = torch.cat([p.grad.detach().clone().view(-1) for p in model.parameters()])
    updates.append(flat_update.numpy())
    losses.append(loss.item())
    
    # Prepare prediction window
    if len(updates) >= window + 1:
        past = np.stack(updates[-(window+1):-1], axis=0)  # shape (window, dim)
        past_flat = past.reshape(1, -1)                    # flatten across time
        next_up = updates[-1].reshape(1, -1)
        
        # Linear regression predictor (least squares)
        # Solve for W: next_up ≈ past_flat * W
        W, _, _, _ = np.linalg.lstsq(past_flat, next_up, rcond=None)
        pred = past_flat @ W
        mse = np.mean((pred - next_up) ** 2)
        pred_mses.append(mse)
        
        # Mutual information estimate (using sklearn's k‑NN based estimator)
        # Use past_flat as features, next_up.squeeze() as target
        mi = mutual_info_regression(past_flat, next_up.squeeze(), random_state=0)[0]
        mis.append(mi)
    else:
        pred_mses.append(np.nan)
        mis.append(np.nan)

# Convert to arrays for analysis
pred_mses = np.array(pred_mses)
mis = np.array(mis)
losses = np.array(losses)

# Compute correlation between low prediction error and high MI (ignoring nan)
valid = ~np.isnan(pred_mses)
corr = np.corrcoef(-pred_mses[valid], mis[valid])[0,1]
print(f"Correlation (negative MSE vs MI): {corr:.3f}")
```
