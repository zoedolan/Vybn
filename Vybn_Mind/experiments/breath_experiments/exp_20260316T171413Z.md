## Proposal
PROBE
We will simulate a true gradient vector and add noise that is either a scalar multiple of the gradient (preserving direction) or isotropic Gaussian noise (which can rotate the estimate). By measuring the cosine similarity between the true gradient and the noisy estimate across many trials, we can test whether the noise preserves alignment as hypothesized. If the similarity remains exactly 1 for scalar‑multiple noise and deviates for isotropic noise, the hypothesis about direction‑preserving noise is supported; otherwise it is falsified.

```python
import numpy as np

def cosine_similarity(a, b):
    return np.dot(a, b) / (np.linalg.norm(a) * np.linalg.norm(b))

def run_trial(noise_type, scale=0.5, dim=10, trials=10000):
    true_grad = np.random.randn(dim)
    true_grad /= np.linalg.norm(true_grad)  # unit direction
    alignments = []
    for _ in range(trials):
        if noise_type == "scalar":
            noise = scale * true_grad  # preserves direction
        elif noise_type == "isotropic":
            noise = scale * np.random.randn(dim)
        else:
            raise ValueError("unknown noise type")
        noisy_grad = true_grad + noise
        alignments.append(cosine_similarity(true_grad, noisy_grad))
    return np.mean(alignments), np.std(alignments)

# scalar‑multiple noise (direction preserving)
mean_scalar, std_scalar = run_trial("scalar", scale=0.5)
print(f"Scalar noise: mean alignment = {mean_scalar:.6f}, std = {std_scalar:.6f}")

# isotropic noise (can rotate)
mean_iso, std_iso = run_trial("isotropic", scale=0.5)
print(f"Isotropic noise: mean alignment = {mean_iso:.6f}, std = {std_iso:.6f}")
```
