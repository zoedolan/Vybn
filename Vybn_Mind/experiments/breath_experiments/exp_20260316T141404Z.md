## Proposal
COMPARE
We will compare the alignment and convergence speed of natural‑gradient steps that use the instantaneous Fisher information matrix versus an exponentially moving‑average (EMA) smoothed Fisher on a simple quadratic loss. The EMA version implements the “local” curvature idea by letting the preconditioner vary gently across parameter steps, and we will measure whether it yields higher alignment with the true gradient and faster loss reduction than the instantaneous version.
```python
import torch
import numpy as np

# Simple quadratic loss: f(x) = 0.5 x^T A x + b^T x
dim = 5
A = torch.randn(dim, dim)
A = A @ A.T + 0.1 * torch.eye(dim)  # make SPD
b = torch.randn(dim)

def loss_and_grad(x):
    loss = 0.5 * x @ A @ x + b @ x
    grad = A @ x + b
    return loss, grad

def natural_step(x, F, grad, damping=1e-3):
    # Solve (F + damping I) v = grad for v (natural gradient)
    Fd = F + damping * torch.eye(F.shape[0])
    v = torch.linalg.solve(Fd, grad)
    return v

opt_steps = 100
lr = 0.1
ema_decay = 0.9

# Initialise
x_inst = torch.zeros(dim, requires_grad=False)
x_ema = torch.zeros(dim, requires_grad=False)
F_ema = torch.zeros(dim, dim)

align_inst = []
align_ema = []
loss_inst = []
loss_ema = []

for t in range(opt_steps):
    # instantaneous Fisher (outer product of grad)
    loss_inst_t, grad_inst = loss_and_grad(x_inst)
    F_inst = torch.outer(grad_inst, grad_inst)
    ng_inst = natural_step(x_inst, F_inst, grad_inst)
    # alignment
    align = (ng_inst @ grad_inst) / (ng_inst.norm() * grad_inst.norm() + 1e-8)
    align_inst.append(align.item())
    loss_inst.append(loss_inst_t.item())
    # SGD‑style update with natural gradient
    x_inst = x_inst - lr * ng_inst

    # EMA Fisher
    loss_ema_t, grad_ema = loss_and_grad(x_ema)
    F_t = torch.outer(grad_ema, grad_ema)
    F_ema = ema_decay * F_ema + (1 - ema_decay) * F_t
    ng_ema = natural_step(x_ema, F_ema, grad_ema)
    align_ema.append((ng_ema @ grad_ema) / (ng_ema.norm() * grad_ema.norm() + 1e-8).item())
    loss_ema.append(loss_ema_t.item())
    x_ema = x_ema - lr * ng_ema

print("Instantaneous Fisher alignment (mean):", np.mean(align_inst))
print("EMA Fisher alignment (mean):", np.mean(align_ema))
print("Final loss instantaneous:", loss_inst[-1])
print("Final loss EMA:", loss_ema[-1])
```
