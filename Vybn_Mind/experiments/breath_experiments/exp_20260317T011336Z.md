import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.data import DataLoader, TensorDataset
import numpy as np

# ----- synthetic data for Task A -----
def make_data(n_samples=2000, n_features=20, n_classes=3, seed=0):
    torch.manual_seed(seed)
    X = torch.randn(n_samples, n_features)
    # create a simple linear decision boundary with some noise
    true_w = torch.randn(n_features, n_classes)
    logits = X @ true_w + 0.5 * torch.randn(n_samples, n_classes)
    y = logits.argmax(dim=1)
    return X, y

X_train, y_train = make_data(n_samples=1500)
X_test,  y_test  = make_data(n_samples=500, seed=1)

train_loader = DataLoader(TensorDataset(X_train, y_train), batch_size=64, shuffle=True)
test_loader  = DataLoader(TensorDataset(X_test,  y_test),  batch_size=64)

# ----- model: feature extractor + linear classifier (probe) -----
class FeatureExtractor(nn.Module):
    def __init__(self, in_dim=20, hidden=64):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(in_dim, hidden),
            nn.ReLU(),
            nn.Linear(hidden, hidden),
            nn.ReLU()
        )
    def forward(self, x):
        return self.net(x)

# LoRA‑style low‑rank adapter (tiny perturbation)
class LoRAAdapter(nn.Module):
    def __init__(self, dim, rank=4, scale=0.01):
        super().__init__()
        self.rank = rank
        self.scale = scale
        self.A = nn.Parameter(torch.randn(dim, rank) * 0.1)
        self.B = nn.Parameter(torch.randn(rank, dim) * 0.1)
    def forward(self, x):
        # x: (batch, dim) -> add low‑rank perturbation
        return x + self.scale * (x @ self.A @ self.B)

def train_model(model, epochs=5, lr=1e-3):
    optimizer = torch.optim.Adam(model.parameters(), lr=lr)
    criterion = nn.CrossEntropyLoss()
    model.train()
    for _ in range(epochs):
        for xb, yb in train_loader:
            feats = model(xb)
            logits = feats @ model.probe_weight.T + model.probe_bias
            loss = criterion(logits, yb)
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()

def linear_probe_accuracy(feat_extractor, probe_weight, probe_bias):
    feat_extractor.eval()
    correct, total = 0, 0
    with torch.no_grad():
        for xb, yb in test_loader:
            feats = feat_extractor(xb)
            logits = feats @ probe_weight.T + probe_bias
            pred = logits.argmax(dim=1)
            correct += (pred == yb).sum().item()
            total   += yb.size(0)
    return correct / total

# ----- initial model -----
feat_ext = FeatureExtractor()
# attach a temporary linear probe for probing (we'll extract its weights after training)
feat_ext.probe_weight = nn.Parameter(torch.randn(3, 64) * 0.1)
feat_ext.probe_bias   = nn.Parameter(torch.zeros(3))

# Train the feature extractor jointly with the probe (simple baseline)
train_model(feat_ext, epochs=5)

# Baseline probe accuracy before hiatus
probe_acc_before = linear_probe_accuracy(feat_ext,
                                         feat_ext.probe_weight.data,
                                         feat_ext.probe_bias.data)
print(f"Baseline probe accuracy: {probe_acc_before:.4f}")

# ----- Hiatus condition 1: pure idleness (no change) -----
probe_acc_idle = linear_probe_accuracy(feat_ext,
                                       feat_ext.probe_weight.data,
                                       feat_ext.probe_bias.data)
print(f"After idle hiatus: {probe_acc_idle:.4f}")

# ----- Hiatus condition 2: tiny random LoRA perturbation -----
lora = LoRAAdapter(dim=64, rank=4, scale=0.005)  # very small scale
# Apply LoRA to the feature extractor's output during a forward pass only
class PerturbedExtractor(nn.Module):
    def __init__(self, base, lora):
        super().__init__()
        self.base = base
        self.lora = lora
        self.probe_weight = base.probe_weight
        self.probe_bias   = base.probe_bias
    def forward(self, x):
        feats = self.base(x)
        return self.lora(feats)

perturbed_ext = PerturbedExtractor(feat_ext, lora)
probe_acc_lora = linear_probe_accuracy(perturbed_ext,
                                       perturbed_ext.probe_weight.data,
                                       perturbed_ext.probe_bias.data)
print(f"After LoRA hiatus: {probe_acc_lora:.4f}")

# Interpretation: if probe_acc_lora drops significantly below baseline,
# the hiatus channel (LoRA noise) is too noisy to preserve Task A info.

## How to interpret a “significant” drop

Because the synthetic task is very simple (a linear decision boundary in a 20‑D Gaussian input space) and the feature extractor is a small ReLU network, the learned representations are usually highly linearly separable. In practice you’ll often see baseline probe accuracies in the **90 %+** range after just a few epochs of training.

If you add a LoRAAdapter with:

* `rank = 4` (far smaller than the feature dimension = 64)
* `scale = 0.005` (i.e., the perturbation is multiplied by 0.005)

the expected magnitude of the added noise is roughly:

```
‖scale * (x @ A @ B)‖ ≈ scale * ‖x‖ * ‖A‖ * ‖B‖
```

With the default initializations (`A, B ~ N(0, 0.1²)`) and typical feature norms (‖x‖ ≈ √64 ≈ 8), the perturbation magnitude is on the order of `0.005 * 8 * 0.1 * 0.1 ≈ 0.0004`, i.e., **four‑tenths of a percent** of the feature scale. That is usually *too small* to cause a large accuracy drop; you might see a decrease of a few‑tenths of a percent at most.

Therefore:

| Observed drop | Interpretation |
|---------------|----------------|
| **< 0.5 %** (e.g., 92.3 % → 91.9 %) | The LoRA hiatus channel is **not** noisy enough to destroy Task A information; the representation is robust to this level of perturbation. |
| **≈ 1 %–2 %** (e.g., 92.3 % → 90.5 %) | The channel introduces noticeable noise; some task‑relevant directions are being perturbed, but the information is still largely recoverable. |
| **> 2 %** (e.g., 92.3 % → 89.0 % or lower) | The hiatus channel is **too noisy** – the low‑rank random perturbation is corrupting the linear‑probe‑decodable subspace enough to cause a measurable degradation, analogous to losing entanglement in the teleportation analogy. |

If you want to make the test more sensitive, you can increase either the `rank` (e.g., to 8 or 16) or the `scale` (e.g., to 0.01 or 0.02) and re‑run the probe. The point at which accuracy starts to fall sharply gives you an empirical estimate of the “noise tolerance” of the representation.
