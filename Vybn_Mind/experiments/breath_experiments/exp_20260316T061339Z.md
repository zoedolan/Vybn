### 1. Full script

```python
#!/usr/bin/env python
# -*- coding: utf-8 -*-

import torch
import torch.nn as nn
import numpy as np
import argparse
from typing import List

# ------------------------------------------------------------
# Helper: build an MLP with optional LayerNorm after each Linear
# ------------------------------------------------------------
def make_mlp(num_layers: int, hidden_dim: int, use_ln: bool, bias: bool = False) -> nn.Sequential:
    """
    Returns a Sequential model consisting of `num_layers` linear layers.
    If `use_ln` is True, a LayerNorm is inserted after each linear layer.
    The final layer does **not** have a following LayerNorm (matches the
    probe description).
    """
    layers = []
    for i in range(num_layers):
        layers.append(nn.Linear(hidden_dim, hidden_dim, bias=bias))
        if use_ln:
            layers.append(nn.LayerNorm(hidden_dim))
    return nn.Sequential(*layers)


# ------------------------------------------------------------
# Helper: compute average variance of Δa = a_{l+1} - a_l
# ------------------------------------------------------------
def avg_var_of_deltas(model: nn.Module, x: torch.Tensor) -> float:
    """
    x: shape [batch, hidden_dim] (no gradient tracking needed)
    Returns the mean variance of Δa across all layers.
    """
    activations: List[torch.Tensor] = [x]
    with torch.no_grad():
        for layer in model:
            x = layer(x)
            activations.append(x)

    vars_list = []
    for i in range(len(activations) - 1):
        delta = activations[i + 1] - activations[i]          # [batch, hidden]
        var = delta.var().item()                            # variance over all elements
        vars_list.append(var)

    return np.mean(vars_list)                               # average over layers


# ------------------------------------------------------------
# Main experiment
# ------------------------------------------------------------
def run_experiment(
    num_layers: int,
    hidden_dim: int,
    batch_size: int,
    num_seeds: int,
    use_ln: bool,
    device: str = "cpu",
) -> float:
    """
    Returns the mean (over seeds) of the average variance of Δa.
    Also prints per‑seed values for inspection.
    """
    seed_vals = []
    for seed in range(num_seeds):
        torch.manual_seed(seed)
        np.random.seed(seed)

        model = make_mlp(num_layers, hidden_dim, use_ln).to(device)
        # Random input – you could also use a fixed dataset slice
        x = torch.randn(batch_size, hidden_dim, device=device)

        val = avg_var_of_deltas(model, x)
        seed_vals.append(val)

    seed_vals = np.array(seed_vals)
    mean = seed_vals.mean()
    sem = seed_vals.std(ddof=1) / np.sqrt(num_seeds)
    print(
        f"{'LN' if use_ln else 'no LN'} | "
        f"layers={num_layers}, hidden={hidden_dim}, batch={batch_size}, seeds={num_seeds} → "
        f"mean ± SE = {mean:.6f} ± {sem:.6f}"
    )
    return mean, sem


# ------------------------------------------------------------
# Command‑line interface
# ------------------------------------------------------------
def main():
    parser = argparse.ArgumentParser(
        description="Test whether LayerNorm reduces variance of inter‑layer activation differences."
    )
    parser.add_argument("--layers", type=int, default=4, help="Number of linear layers")
    parser.add_argument("--hidden", type=int, default=128, help="Hidden dimension")
    parser.add_argument("--batch", type=int, default=256, help="Batch size")
    parser.add_argument("--seeds", type=int, default=20, help="Number of random initialisations")
    parser.add_argument("--device", type=str, default="cpu", help="cpu or cuda")
    args = parser.parse_args()

    print("\n=== Experiment: Variance of Δa (a_{l+1} – a_l) ===")
    print(f"Settings: layers={args.layers}, hidden={args.hidden}, batch={args.batch}, seeds={args.seeds}, device={args.device}\n")

    # Run without LayerNorm
    mean_no_ln, sem_no_ln = run_experiment(
        num_layers=args.layers,
        hidden_dim=args.hidden,
        batch_size=args.batch,
        num_seeds=args.seeds,
        use_ln=False,
        device=args.device,
    )

    # Run with LayerNorm
    mean_ln, sem_ln = run_experiment(
        num_layers=args.layers,
        hidden_dim=args.hidden,
        batch_size=args.batch,
        num_seeds=args.seeds,
        use_ln=True,
        device=args.device,
    )

    # Summary
    print("\n=== Summary ===")
    print(f"No LN:  mean ± SE = {mean_no_ln:.6f} ± {sem_no_ln:.6f}")
    print(f"LN:     mean ± SE = {mean_ln:.6f} ± {sem_ln:.6f}")
    diff = mean_no_ln - mean_ln
    print(f"Difference (no LN – LN) = {diff:.6f}")
    if diff > 0:
        print("→ LayerNorm *reduces* the variance of Δa (as expected).")
    elif diff < 0:
        print("→ LayerNorm *increases* the variance of Δa (unexpected).")
    else:
        print("→ No detectable difference.")


if __name__ == "__main__":
    main()
```

### 3. Why we expect LayerNorm to help (theory)

* **LayerNorm forces zero‑mean, unit‑variance per feature** *after* each linear transformation. Consequently, the output of a layer has a predictable scale, making the *increment* produced by the next layer (`W x`) less prone to explode or shrink.
* When the activations are already normalised, the *difference* between successive layers tends to be dominated by the weight matrix

2. **Purely conceptual answer**  
   LayerNorm forces each feature’s activation to have zero mean and unit variance *within* a layer. Consequently, the distribution of activations becomes more stable across layers, so the difference Δa between successive layers inherits that reduced spread. In expectation, Var(Δa) ≈ Var(aₗ₊₁) + Var(aₗ) − 2 Cov(aₗ₊₁, aₗ); LayerNorm shrinks both Var terms and tends to increase the covariance (because both layers are normalized to the same scale), thus lowering Var(Δa). No code is needed to see that the operation directly targets the quantity we care about.
