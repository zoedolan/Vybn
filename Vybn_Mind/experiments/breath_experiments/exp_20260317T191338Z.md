```python
import numpy as np

def kuramoto(N, K, sigma, dt=0.01, T=50.0, seed=0):
    """Simulate Kuramoto oscillators with coupling matrix K."""
    rng = np.random.default_rng(seed)
    theta = rng.uniform(0, 2*np.pi, N)
    omega = rng.normal(0, sigma, N)
    steps = int(T/dt)
    R = np.zeros(steps)
    for s in range(steps):
        # pairwise coupling
        coupling = np.dot(K, np.sin(theta[:, None] - theta[None, :])) / N
        dtheta = omega + coupling
        theta += dtheta * dt
        theta = np.mod(theta, 2*np.pi)
        R[s] = np.abs(np.sum(np.exp(1j*theta))) / N
    return R

# Parameters
N = 100
sigma = 0.5
dt = 0.01
T = 30.0

# 1) All-to-all uniform coupling
K_all = np.ones((N, N))
R_all = kuramoto(N, K_all, sigma, dt, T)

# 2) Split network: two groups with stronger intra-group coupling
K_split = np.zeros((N, N))
half = N//2
K_split[:half, :half] = 2.0   # stronger intra-group
K_split[half:, half:] = 2.0
K_split += 0.5                # baseline weak all-to-all
R_split = kuramoto(N, K_split, sigma, dt, T)

# Build histograms (bins)
bins = np.linspace(0, 1, 30)
hist_all, _ = np.histogram(R_all, bins=bins, density=True)
hist_split, _ = np.histogram(R_split, bins=bins, density=True)

# Goal‑Drift Index: symmetric KL divergence (Jensen‑Shannon)
def js_div(p, q):
    p = np.asarray(p); q = np.asarray(q)
    m = 0.5 * (p + q)
    return 0.5 * (np.sum(p * np.log(p/m)) + np.sum(q * np.log(q/m)))

gdi = js_div(hist_all, hist_split)
print(f"GDI (JS divergence) between all-to-all and split topology: {gdi:.4f}")
```
