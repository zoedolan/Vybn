def gmm_em(X, n_components, max_iter=100, tol=1e-4):
    """Fit a Gaussian mixture with diagonal covariances using EM.
    Returns (means, covars, weights, loglik)."""
    N, D = X.shape
    # init
    resp = np.ones((N, n_components)) / n_components
    means = X[np.random.choice(N, n_components, replace=False)]
    covars = np.var(X, axis=0) + 1e-6
    covars = np.tile(covars, (n_components, 1))
    weights = np.ones(n_components) / n_components

    prev_ll = -np.inf
    for _ in range(max_iter):
        # E-step
        log_prob = np.zeros((N, n_components))
        for k in range(n_components):
            diff = X - means[k]
            log_prob[:, k] = np.log(weights[k]) - 0.5 * np.sum(
                (diff ** 2) / covars[k] + np.log(2 * np.pi * covars[k]), axis=1
            )
        resp = np.exp(log_prob - np.max(log_prob, axis=1, keepdims=True))
        resp /= resp.sum(axis=1, keepdims=True)

        # M-step
        Nk = resp.sum(axis=0)
        weights = Nk / N
        means = (resp.T @ X) / Nk[:, None]
        for k in range(n_components):
            diff = X - means[k]
            covars[k] = np.sum(resp[:, k][:, None] * (diff ** 2), axis=0) / Nk[k] + 1e-6

        # log-likelihood
        ll = np.sum(np.log(np.sum(np.exp(log_prob), axis=1)))
        if ll - prev_ll < tol:
            break
        prev_ll = ll
    return means, covars, weights, ll

def bic_score(X, n_components):
    means, covars, weights, ll = gmm_em(X, n_components)
    D = X.shape[1]

    p = n_components * (2 * D) + (n_components - 1)
    return -2 * ll + p * np.log(X.shape[0])

import numpy as np
import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.data import DataLoader, TensorDataset

# ----------------------------------------------------------------------
# Helper: simple GMM EM (numpy) – diagonal covariances
# ----------------------------------------------------------------------
def gmm_em(X, n_components, max_iter=200, tol=1e-4):
    """Fit a Gaussian mixture with diagonal covariances using EM.
    Returns (means, covars, weights, loglik)."""
    N, D = X.shape

    # ---- initialization ------------------------------------------------
    resp = np.ones((N, n_components)) / n_components
    means = X[np.random.choice(N, n_components, replace=False)]
    # start with the empirical variance of each dimension (plus a tiny jitter)
    covars = np.var(X, axis=0) + 1e-6
    covars = np.tile(covars, (n_components, 1))          # shape (K, D)
    weights = np.ones(n_components) / n_components

    prev_ll = -np.inf
    for _ in range(max_iter):
        # ---- E‑step ----------------------------------------------------
        log_prob = np.zeros((N, n_components))
        for k in range(n_components):
            diff = X - means[k]
            # log N(x|μ_k, Σ_k)  with diagonal Σ_k
            log_prob[:, k] = (
                np.log(weights[k])
                - 0.5 * np.sum((diff ** 2) / covars[k] + np.log(2 * np.pi * covars[k]), axis=1)
            )
        # normalize in log‑space for stability
        log_prob -= np.max(log_prob, axis=1, keepdims=True)
        resp = np.exp(log_prob)
        resp /= resp.sum(axis=1, keepdims=True)

        # ---- M‑step ----------------------------------------------------
        Nk = resp.sum(axis=0)                     # (K,)
        weights = Nk / N
        means = (resp.T @ X) / Nk[:, None]        # (K, D)

        for k in range(n_components):
            diff = X - means[k]
            covars[k] = np.sum(resp[:, k][:, None] * (diff ** 2), axis=0) / Nk[k] + 1e-6

        # ---- log‑likelihood --------------------------------------------
        ll = np.sum(np.log(np.sum(np.exp(log_prob), axis=1)))
        if ll - prev_ll < tol:
            break
        prev_ll = ll

    return means, covars, weights, ll


def bic_score(X, n_components):
    """BIC = -2 * loglik + p * log(N)  (p = # free parameters)."""
    _, _, _, ll = gmm_em(X, n_components)
    N, D = X.shape
    # For each component: mean (D) + variance (D) + weight (1)  → 2D+1
    # Because the weights sum to 1 we subtract one weight DOF.
    p = n_components * (2 * D + 1) - 1
    return -2 * ll + p * np.log(N)


# ----------------------------------------------------------------------
# Synthetic data generation to mimic LoRA updates
# ----------------------------------------------------------------------
def generate_updates(n_steps, dim, mix_strength=0.0, seed=0):
    """
    If mix_strength == 0: unimodal Gaussian.
    If mix_strength > 0: mixture of two Gaussians with separation proportional
    to mix_strength along the first coordinate.
    Returns array shape (n_steps, dim).
    """
    rng = np.random.RandomState(seed)
    if mix_strength == 0.0:
        return rng.randn(n_steps, dim)
    else:
        # two modes separated along first dimension
        n1 = n_steps // 2
        n2 = n_steps - n1
        mode1 = rng.randn(n1, dim)
        mode2 = rng.randn(n2, dim)
        mode2[:, 0] += mix_strength * np.sqrt(dim)   # shift in the first dir
        updates = np.vstack([mode1, mode2])
        rng.shuffle(updates)                         # random ordering
        return updates


# ----------------------------------------------------------------------
# Experiment: compare BIC for 1 vs 2 components under varying weak‑supervision strength
# ----------------------------------------------------------------------
def run_experiment():
    dim = 20          # dimensionality of LoRA update vector (flattened)
    n_steps = 2000    # number of training steps we record
    strengths = [0.0, 0.5, 1.0, 2.0]  # increasing weak‑supervision signal

    print(f"{'strength':>8} | {'BIC_1':>8} | {'BIC_2':>8} | {'ΔBIC (2‑1)':>12}")
    print("-" * 45)
    for s in strengths:
        X = generate_updates(n_steps=n_steps, dim=dim, mix_strength=s, seed=42)

        bic1 = bic_score(X, n_components=1)
        bic2 = bic_score(X, n_components=2)

        delta = bic2 - bic1   # negative → 2‑component model wins
        print(f"{s:8.2f} | {bic1:8.1f} | {bic2:8.1f} | {delta:12.1f}")

    # Optional: a quick visual check (requires matplotlib)
    try:
        import matplotlib.pyplot as plt
        for s in strengths:
            X = generate_updates(n_steps=n_steps, dim=dim, mix_strength=s, seed=42)
            plt.scatter(X[:500, 0], X[:500, 1], label=f"strength={s}", alpha=0.6, s=10)
        plt.xlabel("LoRA‑update dim 0")
        plt.ylabel("LoRA‑update dim 1")
        plt.title("Synthetic update clouds (first two dimensions)")
        plt.legend()
        plt.show()
    except Exception as e:
        print("\n(Plotting skipped – matplotlib not available or failed.)")
        print(e)


if __name__ == "__main__":
    run_experiment()

### What the script does

| Step | Purpose |
|------|---------|
| **`gmm_em`** | EM algorithm for a *diagonal‑covariance* Gaussian mixture (fast, works fine for low‑dimensional synthetic data). |
| **`bic_score`** | Computes the Bayesian Information Criterion; lower BIC → better trade‑off between fit and model complexity. |
| **`generate_updates`** | Produces synthetic LoRA‑update vectors. When `mix_strength=0` the data are a single isotropic Gaussian. As `mix_strength` grows, a second mode is shifted along the first dimension by `mix_strength·√dim`. |
