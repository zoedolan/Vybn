## Proposal
PROBE
We will test whether adding isotropic noise to a two‑component order parameter that otherwise enjoys a pure scalar shift invariance shrinks the range over which power‑law scaling holds, mirroring the observed contraction of the critical region with increasing chemical potential. We simulate the order‑parameter fluctuations near a critical point, apply a uniform scalar shift (μ) and an isotropic Gaussian noise component, compute the two‑point correlation function, and extract the scaling exponent and the scale range where a power law fits within a tolerance.

```python
import numpy as np

def simulate_fluctuations(L=128, beta=1.0, mu=0.0, sigma_iso=0.0, n_samples=5000):
    """
    Generate samples of a 2‑component order parameter (x, y) near criticality.
    - L: linear size of the system (not used directly, just for scaling)
    - beta: inverse temperature controlling the width of the Gaussian fluctuations
    - mu: uniform scalar shift added equally to both components (chemical potential analogue)
    - sigma_iso: standard deviation of isotropic noise added independently to each component
    Returns arrays x, y of shape (n_samples,).
    """
    # Base critical fluctuations: each component Gaussian with variance 1/beta
    base = np.random.normal(loc=0.0, scale=np.sqrt(1.0/beta), size=(2, n_samples))
    # Apply uniform scalar shift
    base[0] += mu
    base[1] += mu
    # Add isotropic noise
    base[0] += np.random.normal(loc=0.0, scale=sigma_iso, size=n_samples)
    base[1] += np.random.normal(loc=0.0, scale=sigma_iso, size=n_samples)
    return base[0], base[1]

def correlation_function(x, y, max_lag=30):
    """Compute the equal‑time correlation C(r) = ⟨(x_i - x_{i+r})(y_i - y_{i+r})⟩."""
    N = len(x)
    C = np.zeros(max_lag)
    for r in range(max_lag):
        if r == 0:
            C[r] = np.mean((x - np.mean(x)) * (y - np.mean(y)))
        else:
            C[r] = np.mean((x[:-r] - np.mean(x[:-r])) * (y[r:] - np.mean(y[r:])))
    return C

def fit_power_law(r, C, r_fit_min, r_fit_max):
    """Fit log C = -alpha * log r + const over [r_fit_min, r_fit_max]."""
    mask = (r >= r_fit_min) & (r <= r_fit_max)
    if np.sum(mask) < 2:
        return None, None
    A = np.vstack([np.log(r[mask]), np.ones(np.sum(mask))]).T
    alpha, const = np.linalg.lstsq(A, np.log(C[mask]), rcond=None)[0]
    return alpha, const

# Parameters
L = 128
beta = 1.0          # critical temperature inverse
mu_vals = [0.0, 0.2, 0.5, 1.0]   # increasing chemical potential (scalar shift)
sigma_iso_vals = [0.0, 0.1, 0.3, 0.5]  # isotropic noise strength

results = {}
for mu in mu_vals:
    for sigma in sigma_iso_vals:
        x, y = simulate_fluctuations(mu=mu, sigma_iso=sigma, n_samples=20000)
        C = correlation_function(x, y, max_lag=50)
        r = np.arange(len(C))
        # Determine scaling range: look for largest contiguous interval where fit error < 5%
        best_range = (0, 0)
        best_alpha = 0
        for rmin in range(1, 20):
            for rmax in range(rmin+5, 40):
                alpha, const = fit_power_law(r, C, rmin, rmax)
                if alpha is None:
                    continue
                pred = const - alpha * np.log(r[rmin:rmax+1])
                err = np.sqrt(np.mean((np.log(C[rmin:rmax+1]) - pred)**2))
                if err < 0.05 and (rmax - rmin) > (best_range[1] - best_range[0]):
                    best_range = (rmin, rmax)
                    best_alpha = alpha
        results[(mu, sigma)] = (best_alpha, best_range[1]-best_range[0])

# Print summary: scaling exponent and width of scaling region
for (mu, sigma), (alpha, width) in results.items():
    print(f"μ={mu:.2f}, σ_iso={sigma:.2f}: α={alpha:.3f}, scaling width={width}")
```
