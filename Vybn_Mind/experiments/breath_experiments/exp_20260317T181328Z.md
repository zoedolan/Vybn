import numpy as np

def simulate_winfree(phi, N=100, K=1.0, omega_range=(0, 2*np.pi),
                     T=2000, dt=0.1, transient=500):
    """Simple Kuramoto‑Winfree simulation with frustration phi."""
    np.random.seed(0)  # reproducibility
    theta = np.random.uniform(0, 2*np.pi, N)
    omega = np.random.uniform(*omega_range, N)
    steps = int(T / dt)
    R_vals = []
    for _ in range(steps):
        # pairwise coupling term
        coupling = (K / N) * np.sum(np.sin(theta[:, None] - theta[None, :] - phi), axis=1)
        theta += (omega + coupling) * dt
        theta %= 2*np.pi
        if _ * dt >= transient:
            R = np.abs(np.sum(np.exp(1j * theta))) / N
            R_vals.append(R)
    return np.array(R_vals)

def bimodality_coefficient(data):
    """Compute bimodality coefficient b = (skew^2 + 1) / kurt."""
    mean = np.mean(data)
    var = np.var(data)
    if var == 0:
        return 0.0
    skew = np.mean(((data - mean) / np.sqrt(var))**3)
    kurt = np.mean(((data - mean) / np.sqrt(var))**4)
    return (skew**2 + 1) / kurt

phis = [0.0, 0.5, 1.0, 1.5]
results = []
for phi in phis:
    R = simulate_winfree(phi)
    hist, bins = np.histogram(R, bins=20, density=False)
    peak_bin = np.argmax(hist)
    bc = bimodality_coefficient(R)
    results.append((phi, peak_bin, bc, hist.copy()))
    print(f"φ={phi:.2f}: peak bin={peak_bin}, bimodality coeff={bc:.3f}")

# Optional: show histogram counts for each φ
