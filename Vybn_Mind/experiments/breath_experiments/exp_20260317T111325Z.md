## Proposal
PROBE
We will simulate a simple LoRA‑style fine‑tuning loop on a synthetic regression task, collect the parameter update vectors Δθ and a frustration penalty defined as the deviation of the current gradient from its exponential moving average. When the penalty exceeds a threshold we temporarily reduce the learning rate (the “drive”). We will then compare the training loss and gradient variance between a baseline with fixed LR and the adaptive‑drive condition to see if the drive yields smoother, more unitary‑like updates.

```python
import numpy as np

# Synthetic data: y = W_true @ x + noise
np.random.seed(42)
dim_in, dim_out = 20, 5
N_samples = 500
X = np.random.randn(N_samples, dim_in)
W_true = np.random.randn(dim_out, dim_in) * 0.1
Y = X @ W_true.T + 0.05 * np.random.randn(N_samples, dim_out)

# LoRA‑like low‑rank adapters: A (rank r) and B such that ΔW = B @ A
rank = 4
A = np.random.randn(rank, dim_in) * 0.01
B = np.random.randn(dim_out, rank) * 0.01

# Optimizer hyper‑params
lr_base = 1e-2
lr_low  = lr_base * 0.2   # drive reduces LR when frustration high
beta = 0.9                # EMA decay for gradient
threshold = 0.5           # frustration threshold
steps = 300
batch_size = 32

def frustration_penalty(grad, grad_ema):
    # normalized deviation of current grad from EMA
    return np.linalg.norm(grad - grad_ema) / (np.linalg.norm(grad_ema) + 1e-8)

# Storage for analysis
delta_thetas = []      # flattened Δθ per step (A,B updates)
frustrations = []
grad_ema = None

loss_hist = []
grad_var_hist = []

for step in range(steps):
    # Mini‑batch
    idx = np.random.choice(N_samples, batch_size, replace=False)
    xb = X[idx]
    yb = Y[idx]

    # Forward pass with current LoRA adaptation
    W = W_true + B @ A          # effective weight
    preds = xb @ W.T
    loss = np.mean((preds - yb) ** 2)
    loss_hist.append(loss)

    # Gradient w.r.t. A and B (simple MSE gradient)
    dL_dpred = 2 * (preds - yb) / batch_size          # (batch, out)
    dL_dW = dL_dpred.T @ xb                           # (out, in)
    # Gradients for low‑rank factors (chain rule)
    grad_A = B.T @ dL_dW                               # (rank, in)
    grad_B = dL_dW @ A.T                               # (out, rank)

    # EMA of gradient (concatenate for simplicity)
    grad_vec = np.concatenate([grad_A.ravel(), grad_B.ravel()])
    if grad_ema is None:
        grad_ema = grad_vec.copy()
    else:
        grad_ema = beta * grad_ema + (1 - beta) * grad_vec

    # Frustration penalty
    frustr = frustration_penalty(grad_vec, grad_ema)
    frustrations.append(frustr)

    # Drive: lower LR if frustration high
    lr = lr_low if frustr > threshold else lr_base

    # Parameter update (SGD)
    A -= lr * grad_A
    B -= lr * grad_B

    # Record update vector (flattened)
    delta_thetas.append(np.concatenate([A.ravel(), B.ravel()]).copy())

    # Track gradient variance as proxy for “non‑unitarity”
    grad_var_hist.append(np.var(grad_vec))

# Convert to arrays for inspection
delta_thetas = np.array(delta_thetas)
frustrations = np.array(frustrations)

print(f"Final loss (adaptive drive): {loss_hist[-1]:.5f}")
print(f"Mean frustration: {frustrations.mean():.3f}")
print(f"Gradient variance mean: {np.mean(grad_var_hist):.5f}")

# Optional: simple check if high‑frustration steps have distinct update direction
high_idx = frustrations > np.percentile(frustrations, 75)
low_idx  = frustrations < np.percentile(frustrations, 25)
if high_idx.any() and low_idx.any():
    # cosine similarity between mean update vectors of the two regimes
    mean_high = delta_thetas[high_idx].mean(axis=0)
    mean_low  = delta_thetas[low_idx].mean(axis=0)
    cos_sim = np.dot(mean_high, mean_low) / (np.linalg.norm(mean_high)*np.linalg.norm(mean_low)+1e-8)
    print(f"Cosine similarity between high/low frustration update means: {cos_sim:.3f}")
```

This script implements the proposed “drive” mechanism, records the needed data, and provides quantitative diagnostics (loss, frustration, gradient variance, and update‑direction similarity) to evaluate whether using the frustration penalty as a control field yields more stable, unitary‑like parameter updates.
