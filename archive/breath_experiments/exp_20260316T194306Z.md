## Proposal
PROBE
We will implement a `shift` method that adds a scalar perturbation (isotropic noise or constant shift) to the optimizer's parameters, then measure the invariant quantity *y − x* (difference between model output and target) before and after the perturbation across training steps. If the invariant truly persists, its statistics should remain unchanged despite the proxy perturbation.

```python
import torch
import torch.nn as nn
import torch.optim as optim

# Simple linear model: y = w * x + b
model = nn.Linear(1, 1)
optimizer = optim.SGD(model.parameters(), lr=0.01)

# Monkey‑patch a shift method onto the optimizer
def shift_optimizer(opt, scale=0.1):
    """Add isotropic Gaussian noise scaled by `scale` to each parameter."""
    with torch.no_grad():
        for p in opt.param_groups[0]['params']:
            p.add_(torch.randn_like(p) * scale)

optimizer.shift = lambda scale=0.1: shift_optimizer(optimizer, scale)

# Dummy dataset: y_true = 2 * x + 0
torch.manual_seed(0)
x = torch.randn(20, 1)
y_true = 2 * x

# Track invariant before/after shift
invariant_before = []
invariant_after = []

for epoch in range(5):
    model.train()
    optimizer.zero_grad()
    y_pred = model(x)
    loss = nn.MSELoss()(y_pred, y_true)
    loss.backward()
    optimizer.step()

    # Compute invariant (y_pred - y_true) before perturbation
    inv = (y_pred - y_true).detach()
    invariant_before.append(inv.mean().item())

    # Apply proxy perturbation via shift
    optimizer.shift(scale=0.05)

    # Re‑evaluate invariant after perturbation (no gradient step)
    with torch.no_grad():
        y_pred_shift = model(x)
        inv_shift = (y_pred_shift - y_true).detach()
        invariant_after.append(inv_shift.mean().item())

print("Invariant before shift:", invariant_before)
print("Invariant after shift:", invariant_after)
```
