import numpy as np

def kuramoto_step(theta, K, dt=0.01):
    """One Euler step of the Kuramoto model with coupling matrix K."""
    N = len(theta)
    coupling = K @ np.sin(theta[:, None] - theta[None, :])  # NxN
    dtheta = 1.0 + np.sum(coupling, axis=1)  # natural frequency set to 1
    return theta + dt * dtheta

def phase_distribution(theta, bins=50):
    """Histogram of phases on [0, 2π)."""
    hist, _ = np.histogram(theta, bins=bins, range=(0, 2*np.pi), density=True)
    return hist

def gdi_from_hist(h1, h2):
    """Simple GDI: symmetrized KL divergence between two histograms."""
    eps = 1e-12
    h1 = np.clip(h1, eps, None)
    h2 = np.clip(h2, eps, None)
    kl12 = np.sum(h1 * np.log(h1 / h2))
    kl21 = np.sum(h2 * np.log(h2 / h1))
    return 0.5 * (kl12 + kl21)

def chsh_statistic(theta, settings):
    """
    Compute CHSH S from phase binary outcomes.
    settings: list of four phase shifts [a, a', b, b'].
    Outcome = +1 if cos(theta + shift) > 0 else -1.
    """
    a, ap, b, bp = settings
    def corr(shift1, shift2):
        x = np.sign(np.cos(theta + shift1))
        y = np.sign(np.cos(theta + shift2))
        return np.mean(x * y)
    S = abs(corr(a, b) + corr(a, bp) + corr(ap, b) - corr(ap, bp))
    return S

N = 200

K0 = (K0 + K0.T) / 2                # make symmetric
theta = np.random.uniform(0, 2*np.pi, N)

# Reference distribution (after a short burn‑in with K0)
for _ in range(100):
    theta = kuramoto_step(theta, K0, dt)
ref_hist = phase_distribution(theta)

# Store results
gdi_vals = []
S_vals = []
perturbations = []

# Apply small random tweaks to K and measure GDI & CHSH
for i in range(30):
    dK = 0.02 * np.random.randn(N, N)
    dK = (dK + dK.T) / 2
    K = K0 + dK
    # let the system relax a bit
    for _ in range(200):
        theta = kuramoto_step(theta, K, dt)
    hist = phase_distribution(theta)
    gdi = gdi_from_hist(hist, ref_hist)
    # choose four measurement settings (phase shifts) arbitrarily
    settings = [0, np.pi/4, np.pi/2, 3*np.pi/4]
    S = chsh_statistic(theta, settings)
    gdi_vals.append(gdi)
    S_vals.append(S)
    perturbations.append(np.linalg.norm(dK))

gdi_vals = np.array(gdi_vals)
S_vals = np.array(S_vals)

print("Fraction of trials with S>2 (CHSH violation):", np.mean(S_vals > 2))
print("Correlation between GDI and S:", np.corrcoef(gdi_vals, S_vals)[0,1])
