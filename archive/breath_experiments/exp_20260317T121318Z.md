We will collect LoRA weight-update vectors (Δθ) tagged by frustration level (high/low) during training, train a simple predictor (e.g., a linear regression model) to forecast the next Δθ from the recent history, and compare the mean‑squared prediction error between the two frustration regimes. A significantly higher error for high‑frustration steps would support the hypothesis that frustration introduces less predictable, steer‑like signals.

```python
import torch
import torch.nn as nn
import numpy as np

# Simulate synthetic data: 10k steps, Δθ in R^20
n_steps = 10000
dim = 20
# Base predictable component: AR(1) with coefficient 0.7
alpha = 0.7
# Frustration label: 0 = low, 1 = high (30% high)
frust = np.random.binomial(1, 0.3, size=n_steps)
# Noise scale higher for high frustration
noise_scale = np.where(frust == 0, 0.1, 0.5)

delta = np.zeros((n_steps, dim))
for t in range(1, n_steps):
    delta[t] = alpha * delta[t-1] + np.random.normal(0, noise_scale[t], size=dim)

# Convert to torch tensors
delta_t = torch.from_numpy(delta).float()
frust_t = torch.from_numpy(frust).long()

# Prepare dataset: predict Δθ_t from Δθ_{t-1}
X = delta_t[:-1]          # features
Y = delta_t[1:]           # targets
F = frust_t[1:]           # frustration label of target step

# Simple linear predictor
model = nn.Linear(dim, dim)
optimizer = torch.optim.SGD(model.parameters(), lr=0.01)
criterion = nn.MSELoss()

# Train on low-frustration subset
low_mask = (F == 0)
X_low, Y_low = X[low_mask], Y[low_mask]
for epoch in range(200):
    optimizer.zero_grad()
    pred = model(X_low)
    loss = criterion(pred, Y_low)
    loss.backward()
    optimizer.step()
low_error = criterion(model(X_low), Y_low).item()

# Test on high-frustration subset (no further training)
high_mask = (F == 1)
X_high, Y_high = X[high_mask], Y[high_mask]
high_error = criterion(model(X_high), Y_high).item()

print(f"Prediction MSE low frustration: {low_error:.5f}")
print(f"Prediction MSE high frustration: {high_error:.5f}")
print(f"Difference (high - low): {high_error - low_error:.5f}")
```
