import torch
import torch.nn as nn
import numpy as np
import matplotlib.pyplot as plt

# -------------------------------------------------
# 1️⃣  Network construction
# -------------------------------------------------
def make_net(use_ln: bool, depth: int = 8, hidden: int = 128) -> nn.Sequential:
    """
    Build a simple feed‑forward net:
        Linear → (optional LayerNorm) → GELU
    repeated `depth` times.
    """
    layers = []
    for _ in range(depth):
        layers.append(nn.Linear(hidden, hidden))
        if use_ln:
            layers.append(nn.LayerNorm(hidden))
        layers.append(nn.GELU())
    return nn.Sequential(*layers)


# -------------------------------------------------
# 2️⃣  Helper: collect activations of every layer
# -------------------------------------------------
def get_activations(net: nn.Sequential, x: torch.Tensor) -> list[torch.Tensor]:
    """
    Returns a list `[a0, a1, ..., aL]` where
        a0 = input (no non‑linearity applied yet)
        a_{i+1} = activation after the i‑th Linear‑(LN?)‑GELU block.
    """
    acts = [x.detach()]               # a0 = raw input
    for module in net:
        x = module(x)
        # we only store after the GELU (i.e. after each block)
        if isinstance(module, nn.GELU):
            acts.append(x.detach())
    return acts


# -------------------------------------------------
# 3️⃣  Variance of successive‑layer differences
# -------------------------------------------------
def variance_of_diffs(
    net: nn.Sequential,
    batch: int = 256,
    trials: int = 50,
    hidden: int = 128,
) -> list[float]:
    """
    For each layer ℓ compute Var[ a_{ℓ+1} – a_ℓ ] over the batch
    and over `trials` random inputs. Returns a list of length `depth`.
    """
    # accumulate squared differences; we'll compute the unbiased variance later
    sum_diff = [torch.zeros(hidden) for _ in range(len(net)//3)]  # one per block
    sum_diff_sq = [torch.zeros(hidden) for _ in range(len(net)//3)]

    with torch.no_grad():
        for _ in range(trials):
            x = torch.randn(batch, hidden)          # fresh random input
            acts = get_activations(net, x)          # [a0, a1, ..., aL]

            # acts[0] is the raw input; acts[1] is after first block, etc.
            for l in range(len(acts) - 1):
                diff = acts[l + 1] - acts[l]        # shape (batch, hidden)
                sum_diff[l] += diff.sum(dim=0)
                sum_diff_sq[l] += (diff ** 2).sum(dim=0)

    # unbiased sample variance:  Var = (E[x^2] - E[x]^2) / (N-1)
    N = batch * trials
    vars_per_layer = []
    for s, ss in zip(sum_diff, sum_diff_sq):
        mean = s / N
        mean_sq = ss / N
        var = (mean_sq - mean ** 2) * (N / (N - 1))   # Bessel's correction
        vars_per_layer.append(var.mean().item())     # average over features
    return vars_per_layer


# -------------------------------------------------
# 4️⃣  Run the experiment
# -------------------------------------------------
def run_experiment(
    depth: int = 8,
    hidden: int = 128,
    batch: int = 256,
    trials: int = 50,
    seeds: list[int] = [0, 1, 2, 3, 4],
):
    """
    Returns two dictionaries:
        vars_without_ln[seed] = list of variances per layer (no LN)
        vars_with_ln[seed]    = list of variances per layer (with LN)
    """
    vars_without_ln = {}
    vars_with_ln = {}

    for seed in seeds:
        torch.manual_seed(seed)
        np.random.seed(seed)

        # ---- without LayerNorm ----
        net_no_ln = make_net(use_ln=False, depth=depth, hidden=hidden)
        vars_without_ln[seed] = variance_of_diffs(
            net_no_ln, batch=batch, trials=trials, hidden=hidden
        )

        # ---- with LayerNorm ----
        net_ln = make_net(use_ln=True, depth=depth, hidden=hidden)
        vars_with_ln[seed] = variance_of_diffs(
            net_ln, batch=batch, trials=trials, hidden=hidden
        )

    return vars_without_ln, vars_with_ln


# -------------------------------------------------
# 5️⃣  Visualise results (optional but helpful)
# -------------------------------------------------
def plot_results(vars_without_ln, vars_with_ln):
    depth = len(next(iter(vars_without_ln.values())))

    # average across seeds
    mean_no_ln = np.mean([vars_without_ln[s] for s in vars_without_ln], axis=0)
    mean_ln = np.mean([vars_with_ln[s] for s in vars_with_ln], axis=0)
    std_no_ln = np.std([vars_without_ln[s] for s in vars_without_ln], axis=0)
    std_ln = np.std([vars_with_ln[s] for s in vars_with_ln], axis=0)

    layers = np.arange(1, depth + 1)   # layer 1 … depth

    plt.figure(figsize=(8, 4.5))
    plt.errorbar(layers, mean_no_ln, yerr=std_no_ln, fmt='o-', capsize=3,
                 label='No LayerNorm')
    plt.errorbar(layers, mean_ln, yerr=std_ln, fmt='s-', capsize=3,
                 label='With LayerNorm')
    plt.yscale('log')
    plt.xlabel('Layer index (ℓ)')
    plt.ylabel('Var[ a_{ℓ+1} – a_ℓ ] (log scale)')
    plt.title('Effect of LayerNorm on successive‑layer activation variance')
    plt.legend()
    plt.grid(True, which='both', ls='--', lw=0.5)
    plt.tight_layout()
    plt.show()


# -------------------------------------------------
# 6️⃣  Execute
# -------------------------------------------------
if __name__ == "__main__":
    vars_no, vars_yes = run_experiment()
    plot_results(vars_no, vars_yes)
