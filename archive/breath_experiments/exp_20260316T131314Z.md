We will test whether adding an L2 weight‑decay regularizer to the natural‑gradient update rescues its performance on the LayerNorm‑shift alignment task. We'll train three optimizers—plain SGD, natural gradient (NG), and natural gradient with weight decay (NG+WD)—on the same synthetic dataset and compare the final cosine similarity between the learned weight vector and the true shift direction. If the hypothesis is correct, NG+WD should achieve similarity comparable to SGD, whereas plain NG will remain near chance.

```python
import torch
import torch.nn as nn
import torch.optim as optim
import numpy as np

# Synthetic task: learn a vector v that aligns with a random shift direction s
def make_batch(batch_size=256, dim=64):
    x = torch.randn(batch_size, dim)               # input features
    s = torch.randn(dim)                           # true shift direction (fixed per run)
    s = s / s.norm()
    y = (x * s).sum(dim=1, keepdim=True)           # target = projection onto s
    return x, y, s

dim = 64
num_steps = 2000
lr = 0.1
wd = 1e-4  # weight decay strength

# True shift (fixed for the experiment)
torch.manual_seed(0)
s_true = torch.randn(dim)
s_true = s_true / s_true.norm()

# Model: linear layer without bias
model = nn.Linear(dim, 1, bias=False)
# Initialize weights randomly
nn.init.normal_(model.weight, std=0.01)

def train(optimizer_fn, label):
    # reinitialize model for each run
    model.weight.data.normal_(0, 0.01)
    opt = optimizer_fn(model.parameters())
    losses = []
    for step in range(num_steps):
        x, y, _ = make_batch()
        pred = model(x)
        loss = nn.MSELoss()(pred, y)
        opt.zero_grad()
        loss.backward()
        opt.step()
        if step % 200 == 0:
            losses.append(loss.item())
    # final alignment
    with torch.no_grad():
        v = model.weight.view(-1)
        align = (v @ s_true) / (v.norm() * s_true.norm())
    return align.item(), losses

# Optimizers
sgd_opt = lambda params: optim.SGD(params, lr=lr, momentum=0.0)
ng_opt  = lambda params: optim.SGD(params, lr=lr)  # placeholder; we'll replace with natural grad below
# Simple Fisher‑approximated natural gradient (diagonal)
def natural_grad_step(params, lr):
    for p in params:
        if p.grad is None:
            continue
        # diagonal Fisher ≈ mean of squared grads over a minibatch
        fish = p.grad.detach().pow(2).mean() + 1e-8
        p.data.add_(-lr * p.grad / fish)

def ng_optimizer(params):
    return [{'params': list(params), 'lr': lr, 'step': natural_grad_step}]

# NG with weight decay: apply L2 penalty manually in step
def ng_wd_step(params, lr, weight_decay):
    for p in params:
        if p.grad is None:
            continue
        fish = p.grad.detach().pow(2).mean() + 1e-8
        grad_natural = p.grad / fish
        p.data.add_(-lr * (grad_natural + weight_decay * p.data))

def ng_wd_optimizer(params):
    return [{'params': list(params), 'lr': lr, 'weight_decay': wd,
             'step': lambda p, lr=lr, wd=wd: ng_wd_step(p, lr, wd)}]

# Run experiments
align_sgd, _ = train(sgd_opt, "SGD")
# Natural gradient (plain)
def train_ng():
    model.weight.data.normal_(0, 0.01)
    for step in range(num_steps):
        x, y, _ = make_batch()
        pred = model(x)
        loss = nn.MSELoss()(pred, y)
        model.zero_grad()
        loss.backward()
        natural_grad_step(model.parameters(), lr)
    with torch.no_grad():
        v = model.weight.view(-1)
        return (v @ s_true) / (v.norm() * s_true.norm())
align_ng = train_ng()
# NG + WD
def train_ng_wd():
    model.weight.data.normal_(0, 0.01)
    for step in range(num_steps):
        x, y, _ = make_batch()
        pred = model(x)
        loss = nn.MSELoss()(pred, y)
        model.zero_grad()
        loss.backward()
        ng_wd_step(model.parameters(), lr, wd)
    with torch.no_grad():
        v = model.weight.view(-1)
        return (v @ s_true) / (v.norm() * s_true.norm())
align_ng_wd = train_ng_wd()

print(f"SGD alignment: {align_sgd:.4f}")
print(f"Natural Gradient alignment: {align_ng:.4f}")
print(f"Natural Gradient + WD alignment: {align_ng_wd:.4f}")
```
