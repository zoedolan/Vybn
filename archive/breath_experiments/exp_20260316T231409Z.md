import numpy as np

def separability_metric(X):
    """Return mean distance between class means divided by pooled std dev."""
    # X: list of two arrays, each shape (n_samples, dim)
    mu0, mu1 = np.mean(X[0], axis=0), np.mean(X[1], axis=0)
    cov0, cov1 = np.cov(X[0], rowvar=False), np.cov(X[1], rowvar=False)
    # pooled variance (trace of covariance)
    var_pool = np.trace(cov0) + np.trace(cov1)
    return np.linalg.norm(mu0 - mu1) / np.sqrt(var_pool + 1e-8)

# Simulate latent space for task A: two Gaussian clusters
dim = 64
n_per_class = 200
seed = 42
rng = np.random.seed(seed)
X0 = rng.randn(n_per_class, dim) + 2.0   # class 0 shifted +2
X1 = rng.randn(n_per_class, dim) - 2.0   # class 1 shifted -2
X_before = [X0, X1]

# Simulate lingering weights after reset as a random orthogonal matrix
Q, _ = np.linalg.qr(rng.randn(dim, dim))  # Q is orthogonal
# Apply transformation (simulating forward pass through unchanged weights)
X0_trans = X0 @ Q
X1_trans = X1 @ Q
X_after_transform = [X0_trans, X1_trans]

# Synthetic "computation" step: a small non-linear rotation that mixes classes
def computation_step(X):
    # Apply a random symmetric matrix that slightly overlaps the clusters
    M = rng.randn(dim, dim)
    M = (M + M.T) / 2  # make symmetric
    return [x @ M for x in X]

X_after_comp = computation_step([X0_trans, X1_trans])

# Compute separability metrics
sep_before = separability_metric(X_before)
sep_after_transform = separability_metric(X_after_transform)
sep_after_comp = separability_metric(X_after_comp)

print(f"Separability after applying lingering weights (pre‑computation): {sep_after_transform:.3f}")
print(f"Separability after computation step: {sep_after_comp:.3f}")
