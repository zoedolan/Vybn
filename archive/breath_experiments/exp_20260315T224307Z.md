### Synthetic segment generators

```python
def low_freq_trend(t, amp=0.5, freq=0.02, drift=0.001):
    """Slow sinusoid + linear drift."""
    return amp * np.sin(2*np.pi*freq*t) + drift*t

def high_freq_noise(t, amp=1.0):
    """White‑noise burst."""
    return amp * np.random.randn(len(t))

def periodic_spike(t, spike_amp=3.0, period=20, width=2):
    """Sparse spikes of given width."""
    sig = np.zeros_like(t)
    spike_idx = np.arange(0, len(t), period)
    for i in spike_idx:
        end = min(i+width, len(t))
        sig[i:end] = spike_amp
    return sig
```

### Cosine‑similarity helper (batched)

```python
def cosine_similarity(x, y):
    """
    x: (B, L)   – batch of windows
    y: (K, Lproto) – prototypes (K=3)
    Returns: (B, K) similarity in [-1,1]
    """
    x_norm = F.normalize(x, p=2, dim=1)          # (B, L)
    y_norm = F.normalize(y, p=2, dim=1)          # (K, Lproto)
    # If L != Lproto we interpolate/truncate to match length.
    if x_norm.shape[1] != y_norm.shape[1]:
        # simple linear interpolation (could also use torch.nn.functional.interpolate)
        x_norm = F.interpolate(x_norm.unsqueeze(1), size=y_norm.shape[1],
                               mode='linear', align_corners=False).squeeze(1)
    return torch.matmul(x_norm, y_norm.T)        # (B, K)
```

## 2. Synthetic Dataset

Each sample = a window of length **L** drawn from a randomly chosen expert.  
Label = expert index (0,1,2).  
We also return the *raw* window so the model can compute similarities internally.

```python
class ExpertWindowDataset(Dataset):
    def __init__(self, n_samples=10_000, min_len=8, max_len=64,
                 seed=0):
        self.n_samples = n_samples
        self.len_range = (min_len, max_len)
        self.rng = np.random.default_rng(seed)
        self.expert_fns = [low_freq_trend, high_freq_noise, periodic_spike]
        self.expert_params = [
            dict(amp=0.5, freq=0.02, drift=0.001),   # low‑freq trend
            dict(amp=1.0),                           # high‑freq noise
            dict(spike_amp=3.0, period=20, width=2)  # periodic spike
        ]

    def __len__(self):
        return self.n_samples

    def __getitem__(self, idx):
        L = self.rng.integers(*self.len_range)
        expert = self.rng.integers(0, 3)          # uniform over experts
        t = np.arange(L)
        params = self.expert_params[expert]
        signal = self.expert_fns[expert](t, **params)
        # Add a tiny observation noise to make the problem non‑trivial
        signal += 0.05 * self.rng.standard_normal(L)
        return torch.from_numpy(signal).float(), torch.tensor(expert, dtype=torch.long), L
```

### 3.1 Shared feature extractor (tiny 1‑D ConvNet)

```python
class FeatureExtractor(nn.Module):
    def __init__(self, in_channels=1, out_dim=32):
        super().__init__()
        self.conv = nn.Sequential(
            nn.Conv1d(in_channels, 16, kernel_size=5, padding=2),
            nn.ReLU(),
            nn.Conv1d(16, out_dim, kernel_size=5, padding=2),
            nn.ReLU(),
            nn.AdaptiveAvgPool1d(1)   # -> (B, out_dim, 1)
        )
        self.out_dim = out_dim

    def forward(self, x):
        # x: (B, 1, L) -> (B, out_dim)
        return self.conv(x).squeeze(-1)
```
