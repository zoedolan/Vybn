## Proposal
PROBE
We will test whether the normalized invariant \(I = (y - x)/(\|x\|+\varepsilon)\) remains stable under isotropic parameter noise but shows a detectable shift when the underlying objective is altered. We'll train a simple linear model on a regression task, then apply two kinds of perturbations: (1) adding small Gaussian noise to the parameters (simulating self‑modification noise) and (2) changing the target function (simulating a genuine objective drift). By comparing the variance of \(I\) across many noise trials to the mean shift induced by the objective change, we can assess if \(I\) serves as a low‑variance signal for drift detection.

```python
import numpy as np

# Simple linear regression: y = W @ x + b
def init_model(input_dim, output_dim):
    W = np.random.randn(output_dim, input_dim) * 0.1
    b = np.zeros(output_dim)
    return W, b

def predict(W, b, X):
    return W @ X.T + b[:, None]  # shape (output_dim, n_samples)

def loss(W, b, X, Y):
    Y_pred = predict(W, b, X)
    return np.mean((Y_pred - Y) ** 2)

def compute_I(W_old, b_old, W_new, b_new, eps=1e-8):
    # flatten parameters into vectors
    x = np.concatenate([W_old.ravel(), b_old.ravel()])
    y = np.concatenate([W_new.ravel(), b_new.ravel()])
    return (y - x) / (np.linalg.norm(x) + eps)

# Settings
input_dim, output_dim = 5, 1
n_samples = 100
X = np.random.randn(n_samples, input_dim)
# True underlying objective: target = 2 * x[:,0] + noise
true_W = np.array([[2.0, 0, 0, 0, 0]])
true_b = np.array([0.0])
Y = predict(true_W, true_b, X) + 0.01 * np.random.randn(n_samples)

# Initialize model
W, b = init_model(input_dim, output_dim)

# Train a few steps with SGD to get a reasonable starting point
lr = 0.01
for _ in range(200):
    grad_W = 2 * ((predict(W, b, X) - Y) @ X) / n_samples
    grad_b = 2 * np.mean(predict(W, b, X) - Y)
    W -= lr * grad_W
    b -= lr * grad_b

# Baseline I (should be near zero because we compare same params)
I_baseline = compute_I(W, b, W, b)
print(f"Baseline I norm: {np.linalg.norm(I_baseline):.2e}")

# --- Test 1: isotropic noise (should keep I small) ---
noise_levels = [1e-4, 1e-3, 1e-2]
I_norms_noise = []
for sigma in noise_levels:
    I_vals = []
    for _ in range(50):
        W_noise = W + sigma * np.random.randn(*W.shape)
        b_noise = b + sigma * np.random.randn(*b.shape)
        I = compute_I(W, b, W_noise, b_noise)
        I_vals.append(np.linalg.norm(I))
    I_norms_noise.append(np.mean(I_vals))
    print(f"Noise sigma={sigma:.0e}: mean |I| = {np.mean(I_vals):.2e}")

# --- Test 2: objective shift (change target) ---
# new target: double the coefficient
new_true_W = np.array([[4.0, 0, 0, 0, 0]])
new_true_b = np.array([0.0])
# Take one SGD step towards new objective to simulate drift
grad_W_new = 2 * ((predict(W, b, X) - (new_true_W @ X.T + new_true_b)) @ X) / n_samples
grad_b_new = 2 * np.mean(predict(W, b, X) - (new_true_W @ X.T + new_true_b))
W_shift = W - lr * grad_W_new
b_shift = b - lr * grad_b_new
I_shift = compute_I(W, b, W_shift, b_shift)
print(f"After objective shift: |I| = {np.linalg.norm(I_shift):.2e}")

# Summary
print("\nSummary:")
print("Noise-induced |I| values:", I_norms_noise)
print("Shift-induced |I|:", np.linalg.norm(I_shift))
```
