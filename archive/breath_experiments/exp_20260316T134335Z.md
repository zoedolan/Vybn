We will test whether using an exponentially moving average (EMA) of the Fisher information matrix improves the alignment of natural‑gradient steps with the true descent direction compared to using the instantaneous Fisher estimate alone. We'll simulate a time‑varying quadratic loss where the curvature matrix drifts slowly, compute noisy Fisher estimates at each step, and measure the cosine similarity between the update direction and the exact natural‑gradient direction over many iterations.
```python
import torch
import numpy as np

# Settings
dim = 20
steps = 500
ema_alpha = 0.1  # EMA update rate
noise_std = 0.5  # noise added to Fisher estimate

# True time‑varying curvature: slowly rotating positive‑definite matrix
def true_curvature(t):
    angle = 0.01 * t
    c, s = np.cos(angle), np.sin(angle)
    R = np.array([[c, -s, *np.zeros((dim-2,))],
                  [s,  c, *np.zeros((dim-2,))],
                  *[np.zeros((dim-2, dim)) for _ in range(dim-2)]])
    # Fill remaining dimensions with identity
    R = np.eye(dim)
    R[:2, :2] = [[c, -s], [s, c]]
    eig = np.linspace(1.0, 5.0, dim)
    return R @ np.diag(eig) @ R.T

# Simulated parameter vector
x = torch.randn(dim, requires_grad=False)

# Storage for alignment metrics
align_inst = []  # instantaneous Fisher
align_ema = []   # EMA Fisher

# EMA accumulator
F_ema = np.eye(dim)

for t in range(steps):
    # True curvature and loss gradient
    A = true_curvature(t)
    grad = torch.from_numpy(A @ x.numpy()).float()
    
    # Noisy instantaneous Fisher estimate (inverse of curvature + noise)
    F_inst = np.linalg.inv(A) + noise_std * np.random.randn(dim, dim)
    F_inst = (F_inst + F_inst.T) / 2  # symmetrize
    
    # Update EMA estimate
    F_ema = ema_alpha * F_inst + (1 - ema_alpha) * F_ema
    
    # Natural‑gradient directions (preconditioned gradient)
    ng_inst = -torch.from_numpy(np.linalg.solve(F_inst, grad.numpy())).float()
    ng_ema  = -torch.from_numpy(np.linalg.solve(F_ema,  grad.numpy())).float()
    
    # Exact natural‑gradient direction using true curvature
    ng_true = -torch.from_numpy(np.linalg.solve(A, grad.numpy())).float()
    
    # Cosine similarity (alignment)
    def align(a, b):
        return torch.dot(a, b) / (torch.norm(a) * torch.norm(b) + 1e-8)
    
    align_inst.append(align(ng_inst, ng_true).item())
    align_ema.append(align(ng_ema,  ng_true).item())

# Report average alignment over last half of training
def avg_last(arr, frac=0.5):
    start = int(len(arr) * (1 - frac))
    return np.mean(arr[start:])

print(f"Average alignment (instantaneous Fisher): {avg_last(align_inst):.3f}")
print(f"Average alignment (EMA Fisher):          {avg_last(align_ema):.3f}")
```
