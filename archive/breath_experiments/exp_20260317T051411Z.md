import torch
import torch.nn as nn
import torch.nn.functional as F
from torchvision import datasets, transforms
from torch.utils.data import DataLoader
import numpy as np
from scipy.stats import pearsonr

# ----- LoRA layer -----
class LoRALinear(nn.Module):
    def __init__(self, in_features, out_features, r=4, alpha=1):
        super().__init__()
        self.linear = nn.Linear(in_features, out_features, bias=False)
        self.lora_A = nn.Parameter(torch.randn(r, in_features) * 0.01)
        self.lora_B = nn.Parameter(torch.zeros(out_features, r))
        self.r = r
        self.alpha = alpha
        self.scaling = alpha / r

    def forward(self, x):
        return self.linear(x) + (x @ self.lora_A.T @ self.lora_B.T) * self.scaling

    def delta(self):
        # Δ = (lora_B @ lora_A) * scaling
        return (self.lora_B @ self.lora_A) * self.scaling

# ----- Simple model -----
class Net(nn.Module):
    def __init__(self):
        super().__init__()
        self.fc1 = LoRALinear(28*28, 256, r=4)
        self.fc2 = nn.Linear(256, 10)

    def forward(self, x):
        x = x.view(-1, 28*28)
        x = F.relu(self.fc1(x))
        return self.fc2(x)

    def get_delta(self):
        return self.fc1.delta()

# ----- Data -----
transform = transforms.Compose([transforms.ToTensor()])
train_set = datasets.MNIST('.', train=True, download=True, transform=transform)
val_set   = datasets.MNIST('.', train=False, download=True, transform=transform)
train_loader = DataLoader(train_set, batch_size=128, shuffle=True)
val_loader   = DataLoader(val_set, batch_size=256)

device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
model = Net().to(device)
optimizer = torch.optim.Adam(model.parameters(), lr=1e-3)
criterion = nn.CrossEntropyLoss()

# ----- Tracking -----
fits = []
accs = []

def subspace_fit(delta, k):
    # delta: matrix (out_features x in_features)
    U, S, Vt = torch.svd(delta)
    total = (S ** 2).sum()
    if total == 0:
        return 0.0
    return (S[:k] ** 2).sum() / total

model.train()
for epoch in range(3):  # short run for demo
    for xb, yb in train_loader:
        xb, yb = xb.to(device), yb.to(device)
        optimizer.zero_grad()
        logits = model(xb)
        loss = criterion(logits, yb)
        loss.backward()
        optimizer.step()

        # compute subspace‑fit of Δ
        delta = model.get_delta().detach().cpu()
        fit = subspace_fit(delta, k=model.fc1.r).item()
        fits.append(fit)

        # validation accuracy
        model.eval()
        correct = total = 0
        with torch.no_grad():
            for xv, yv in val_loader:
                xv, yv = xv.to(device), yv.to(device)
                preds = model(xv).argmax(dim=1)
                correct += (preds == yv).eq(1).sum().item()
                total   += yv.size(0)
        acc = correct / total
        accs.append(acc)
        model.train()

# correlation
corr, pval = pearsonr(fits, accs)
print(f"Pearson r = {corr:.4f} (p = {pval:.4f})")
