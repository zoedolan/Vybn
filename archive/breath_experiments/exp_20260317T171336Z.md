import numpy as np

def winfree_hebbian(N=100, K0=0.5, eta=0.01, dt=0.05, T=200.0, phi=0.0, seed=42):
    np.random.seed(seed)
    # natural frequencies
    omega = np.random.normal(0.0, 0.5, N)
    # phases
    theta = np.random.uniform(0, 2*np.pi, N)
    # coupling matrix (symmetric, zero diagonal)
    K = np.full((N, N), K0)
    np.fill_diagonal(K, 0.0)
    
    steps = int(T/dt)
    R_hist = []
    
    for s in range(steps):
        # compute pairwise phase differences
        diff = theta[:, None] - theta[None, :]  # NxN
        # Kuramoto‑Winfree coupling with frustration
        coupling = (K * np.sin(diff - phi)).sum(axis=1) / N
        theta = theta + (omega + coupling) * dt
        # keep phases in [0, 2π)
        theta = np.mod(theta, 2*np.pi)
        
        # Hebbian adaptation of couplings
        cos_diff = np.cos(diff)
        dK = eta * (cos_diff - K)
        K = K + dK * dt
        np.fill_diagonal(K, 0.0)
        
        # order parameter
        R = np.abs(np.sum(np.exp(1j*theta))) / N
        R_hist.append(R)
    
    return np.array(R_hist)
