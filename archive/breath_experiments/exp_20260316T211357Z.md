```python
import torch
import torch.nn as nn
import torch.nn.functional as F
import numpy as np

# ----- A minimal vision‑language model (placeholder) -----
class TinyVLM(nn.Module):
    def __init__(self, img_dim=64, txt_dim=32, hid_dim=128):
        super().__init__()
        self.img_enc = nn.Linear(img_dim, hid_dim)
        self.txt_enc = nn.Linear(txt_dim, hid_dim)
        self.fusion  = nn.Linear(hid_dim*2, hid_dim)
        self.head    = nn.Linear(hid_dim, 10)   # dummy logits (e.g., answer classes)

    def forward(self, img, txt):
        i = self.img_enc(img)
        t = self.txt_enc(txt)
        x = torch.cat([i, t], dim=-1)
        h = F.relu(self.fusion(x))
        return self.head(h), h   # return logits and hidden representation

# ----- Synthetic data: near‑identical images & synonymous questions -----
def make_pair(batch_size=1):
    # image: small random noise; second image differs by tiny epsilon
    img1 = torch.randn(batch_size, 64)
    img2 = img1 + 1e-3 * torch.randn_like(img1)   # near‑identical
    # question embeddings: synonyms -> same vector
    txt = torch.randn(batch_size, 32)
    return (img1, txt), (img2, txt)

model = TinyVLM()
model.eval()

# ----- Experiment: add Gaussian noise to parameters -----
def add_param_noise(model, sigma):
    with torch.no_grad():
        for p in model.parameters():
            p.add_(torch.randn_like(p) * sigma)

def compute_metrics(img_txt_pair1, img_txt_pair2, sigma, eps=1e-8):
    add_param_noise(model, sigma)
    with torch.no_grad():
        logits1, hid1 = model(*img_txt_pair1)
        logits2, hid2 = model(*img_txt_pair2)

    # raw difference (L2 norm of difference)
    raw_diff = (hid1 - hid2).norm(p=2, dim=-1)   # shape (batch,)
    # normalized difference: ||h1-h2|| / (||h1|| + ||h2|| + eps)
    norm_diff = (hid1 - hid2).norm(p=2, dim=-1) / (hid1.norm(p=2, dim=-1) + hid2.norm(p=2, dim=-1) + eps)

    return raw_diff.item(), norm_diff.item()

# ----- Run over many noise seeds -----
sigmas = [0.0, 1e-4, 1e-3, 1e-2]
n_seeds = 200
results = {s: {"raw": [], "norm": []} for s in sigmas}

for sigma in sigmas:
    for seed in range(n_seeds):
        torch.manual_seed(seed)
        # reset model to original weights each seed
        model.load_state_dict(TinyVLM().state_dict())
        raw, norm = compute_metrics(*make_pair(), sigma)
        results[sigma]["raw"].append(raw)
        results[sigma]["norm"].append(norm)

# ----- Summarize variance -----
for sigma in sigmas:
    raw_var = np.var(results[sigma]["raw"])
    norm_var = np.var(results[sigma]["norm"])
    print(f"σ={sigma:.0e}: raw var={raw_var:.2e}, norm var={norm_var:.2e}, ratio={raw_var/norm_var:.1f}")
```
