## Proposal
PROBE
We will train a small language model with LoRA adapters while logging a frustration penalty defined as the exponential moving average of the loss deviation from a running baseline. After training, we collect the LoRA weight‑update vectors (Δθ) for each optimization step and fit a Gaussian Mixture Model with two components. We then test whether the posterior probability of the high‑frustration component correlates strongly with steps where the frustration penalty exceeds a threshold, indicating that the penalty acts as an internal prompt driving the model into a distinct update regime.

```python
import torch
import torch.nn as nn
from torch.utils.data import DataLoader, Dataset
from sklearn.mixture import GaussianMixture
import numpy as np

# ---- Dummy dataset ----
class RandomTextDataset(Dataset):
    def __init__(self, size=1000, vocab_size=500, seq_len=20):
        self.size = size
        self.vocab_size = vocab_size
        self.seq_len = seq_len
        self.data = torch.randint(0, vocab_size, (size, seq_len))

    def __len__(self):
        return self.size

    def __getitem__(self, idx):
        return self.data[idx]

# ---- Simple model with LoRA ----
class LoRALayer(nn.Module):
    def __init__(self, in_dim, out_dim, rank=4):
        super().__init__()
        self.weight = nn.Parameter(torch.randn(out_dim, in_dim) * 0.01)
        self.lora_A = nn.Parameter(torch.randn(rank, in_dim) * 0.01)
        self.lora_B = nn.Parameter(torch.zeros(out_dim, rank))
        self.rank = rank

    def forward(self, x):
        # x: (batch, in_dim)
        lora_update = (self.lora_B @ self.lora_A)  # (out_dim, in_dim)
        return x @ (self.weight + lora_update).t()

class TinyModel(nn.Module):
    def __init__(self, vocab_size=500, embed_dim=64, hidden_dim=128):
        super().__init__()
        self.embed = nn.Embedding(vocab_size, embed_dim)
        self.lora = LoRALayer(embed_dim, hidden_dim)
        self.fc = nn.Linear(hidden_dim, vocab_size)

    def forward(self, input_ids):
        x = self.embed(input_ids)          # (B, L, D)
        x = x.mean(dim=1)                  # simple pooling
        x = self.lora(x)                   # (B, H)
        logits = self.fc(x)                # (B, V)
        return logits

# ---- Training loop with frustration penalty ----
device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
model = TinyModel().to(device)
optimizer = torch.optim.Adam(model.parameters(), lr=1e-3)
criterion = nn.CrossEntropyLoss()

dataset = RandomTextDataset()
loader = DataLoader(dataset, batch_size=32, shuffle=True)

frustration_alpha = 0.9  # EMA weight
frustration_ema = 0.0
loss_baseline = 0.0

updates = []      # store flattened LoRA Δθ per step
frustrations = [] # corresponding frustration value

for epoch in range(3):
    for batch in loader:
        batch = batch.to(device)
        optimizer.zero_grad()
        logits = model(batch)
        loss = criterion(logits, batch[:, -1])  # predict next token
        loss.backward()
        optimizer.step()

        # compute frustration as EMA of loss deviation
        loss_val = loss.item()
        frustration_ema = frustration_alpha * frustration_ema + (1 - frustration_alpha) * abs(loss_val - loss_baseline)
        loss_baseline = 0.9 * loss_baseline + 0.1 * loss_val  # slowly moving baseline

        # capture LoRA update (θ_before - θ_after) – we approximate by the gradient step
        with torch.no_grad():
            delta = model.lora.lora_B.grad.clone().detach().cpu().numpy().ravel()
            updates.append(delta)
            frustrations.append(frustration_ema)

updates = np.stack(updates)          # (N_steps, D)
frustrations = np.array(frustrations)

# ---- Fit GMM with 2 components on LoRA updates ----
gmm = GaussianMixture(n_components=2, covariance_type='full', random_state=0)
gmm.fit(updates)
probs = gmm.predict_proba(updates)[:, 1]  # probability of component 1

# ---- Test correlation with high frustration ----
high_frust = frustrations > np.percentile(frustrations, 75)  # top quartile
corr = np.corrcoef(probs, high_frust.astype(float))[0, 1]
print(f"Correlation between high‑frustration indicator and GMM component 1 posterior: {corr:.3f}")
```
