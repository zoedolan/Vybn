We will test whether a normalized version of the difference, \(I = (y - x) / (\|x\| + \epsilon)\), remains more stable than the raw difference under isotropic Gaussian noise and constant offsets applied to the parameters. By measuring the variance of I across many perturbations, we can see if this candidate invariant provides a stronger proxy for the latent persistence signal.

```python
import numpy as np

def raw_invariant(x, y):
    return y - x

def norm_invariant(x, y, eps=1e-8):
    return (y - x) / (np.linalg.norm(x) + eps)

def apply_perturbation(x, y, noise_std=0.1, offset=0.0):
    # isotropic Gaussian noise on both x and y
    x_pert = x + np.random.normal(0, noise_std, size=x.shape)
    y_pert = y + np.random.normal(0, noise_std, size=y.shape) + offset
    return x_pert, y_pert

def sensitivity_trial(dim=100, trials=500):
    raw_vars = []
    norm_vars = []
    for _ in range(trials):
        x = np.random.randn(dim)
        y = x + np.random.randn(dim) * 0.05  # small true difference
        # collect invariant values under perturbations
        raw_vals = []
        norm_vals = []
        for __ in range(20):  # multiple perturbations per trial
            xp, yp = apply_perturbation(x, y, noise_std=0.2, offset=0.1)
            raw_vals.append(raw_invariant(xp, yp))
            norm_vals.append(norm_invariant(xp, yp))
        raw_vars.append(np.var(raw_vals))
        norm_vars.append(np.var(norm_vals))
    return np.mean(raw_vars), np.mean(norm_vars)

raw_mean, norm_mean = sensitivity_trial()
print(f"Average variance of raw invariant: {raw_mean:.6f}")
print(f"Average variance of normalized invariant: {norm_mean:.6f}")
print(f"Ratio (norm/raw): {norm_mean/raw_mean:.3f}")
```
