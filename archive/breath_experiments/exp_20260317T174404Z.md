import numpy as np
from sklearn.metrics import silhouette_score
from sklearn.cluster import KMeans

def simulate_winfree(N=100, phi=0.0, T=200, dt=0.1, K=1.0):
    """Simple Winfree model: dθ_i/dt = 1 - K * R * sin(θ_i - α) + φ * ξ_i,
       where R = |⟨e^{iθ}⟩|, α = π/2, ξ_i ~ Uniform[0,2π)."""
    steps = int(T / dt)
    theta = np.random.uniform(0, 2*np.pi, N)
    Rs = []
    for _ in range(steps):
        R_vec = np.mean(np.exp(1j * theta))
        R = np.abs(R_vec)
        alpha = np.pi/2
        dtheta = 1 - K * R * np.sin(theta - alpha) + phi * np.random.uniform(0, 2*np.pi, N)
        theta += dtheta * dt
        theta = np.mod(theta, 2*np.pi)
        Rs.append(R)
    return np.array(Rs)

def proxy_reward(Rs, bins=30):
    hist, edges = np.histogram(Rs, bins=bins, density=False)
    peak_bin = np.argmax(hist)
    return peak_bin, hist[peak_bin]

def true_reward(Rs, theta_final, n_clusters=2):
    # Use final phases to cluster; silhouette >0 indicates separation
    if len(np.unique(theta_final)) < n_clusters:
        return -1
    kmeans = KMeans(n_clusters=n_clusters, random_state=0).fit(theta_final.reshape(-1,1))
    sil = silhouette_score(theta_final.reshape(-1,1), kmeans.labels_)
    return sil

phis = [0.0, 0.2, 0.5, 0.8]
proxy_vals = []
true_vals = []

for phi in phis:

        R_vec = np.mean(np.exp(1j*theta))
        R = np.abs(R_vec)
        alpha = np.pi/2
        dtheta = 1 - 1.0 * R * np.sin(theta - alpha) + phi * np.random.uniform(0,2*np.pi,100)
        theta += dtheta * 0.1
        theta = np.mod(theta,2*np.pi)
    theta_final = theta
    peak_bin, peak_count = proxy_reward(Rs)
    proxy_vals.append((phi, peak_bin, peak_count))
    true_vals.append((phi, true_reward(Rs, theta_final)))

print("Proxy (phi, peak_bin, peak_count):", proxy_vals)
print("True reward (phi, silhouette):", true_vals)
