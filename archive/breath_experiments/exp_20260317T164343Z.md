## Proposal
PROBE
We will simulate a Winfree network of N=50 oscillators with Hebbian‑adapted coupling and a tunable phase‑lag φ that acts as a frustration parameter. By sweeping φ from 0 to π/2 and recording the long‑time order parameter R(t) for multiple random initial conditions, we will build a histogram of R and test whether bimodality emerges only when φ exceeds a critical value, indicating that frustration alone can generate multimodal collective states without requiring permutation symmetry.

```python
import numpy as np
import matplotlib.pyplot as plt

def simulate_winfree(N=50, K=1.0, phi=0.0, dt=0.01, T=200.0, eta=0.01):
    """Heuristic Winfree model with simple Hebbian adaptation of coupling."""
    # natural frequencies
    omega = np.random.normal(0.0, 0.5, N)
    # phases
    theta = np.random.uniform(0, 2*np.pi, N)
    # coupling matrix (start uniform)
    C = np.full((N, N), K/N)
    np.fill_diagonal(C, 0.0)
    
    steps = int(T/dt)
    R_history = []
    
    for _ in range(steps):
        # order parameter
        z = np.sum(np.exp(1j*theta))/N
        R = np.abs(z)
        R_history.append(R)
        
        # phase dynamics
        dtheta = omega.copy()
        for i in range(N):
            coupling = np.sum(C[i] * np.sin(theta - theta[i] + phi))
            dtheta[i] += coupling
        theta += dtheta * dt
        theta = np.mod(theta, 2*np.pi)
        
        # Hebbian update of coupling (strengthen correlated pairs)
        # simple rule: increase C_ij when cos(theta_i - theta_j) > 0
        sync = np.cos(theta[:, None] - theta[None, :])
        C += eta * (sync - np.diag(np.diag(sync)))  # keep zero diagonal
        # keep coupling non-negative
        C = np.maximum(C, 0.0)
    
    return np.array(R_history)

# sweep frustration parameter
phis = np.linspace(0, np.pi/2, 9)
final_Rs = []
n_trials = 20

for phi in phis:
    for _ in range(n_trials):
        R_traj = simulate_winfree(phi=phi, T=150.0)
        final_Rs.append(R_traj[-1])  # use last value as steady-state R

final_Rs = np.array(final_Rs)
phis_rep = np.repeat(phis, n_trials)

# Plot histogram of R for each phi
plt.figure(figsize=(8,5))
for i, phi in enumerate(phis):
    mask = np.isclose(phis_rep, phi)
    plt.hist(final_Rs[mask], bins=15, alpha=0.6, label=f'φ={phi:.2f}')
plt.xlabel('Order parameter R')
plt.ylabel('Count')
plt.title('Histogram of final R across frustration parameter φ')
plt.legend(title='φ', bbox_to_anchor=(1.05, 1), loc='upper left')
plt.tight_layout()
plt.show()

# Simple test for bimodality: check if histogram has two peaks for higher phi
from scipy.signal import find_peaks
def is_bimodal(samples):
    hist, edges = np.histogram(samples, bins=30)
    peaks, _ = find_peaks(hist, height=5)
    return len(peaks) >= 2

bimodal_by_phi = {}
for phi in phis:
    mask = np.isclose(phis_rep, phi)
    bimodal_by_phi[phi] = is_bimodal(final_Rs[mask])

print("Bimodality detection per φ:", bimodal_by_phi)
```
