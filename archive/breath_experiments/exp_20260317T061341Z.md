import torch
import torch.nn as nn
import torch.nn.functional as F
import numpy as np
from sklearn.mixture import GaussianMixture
# Hartigan's dip test implementation (simple version)
def hartigans_dip(data):
    # data: 1D numpy array
    n = len(data)
    sorted_data = np.sort(data)
    # Empirical CDF
    F_n = np.arange(1, n+1) / n
    # Compute the dip statistic as max|F_n - F| where F is the best-fitting unimodal CDF
    # Approximate by checking all possible intervals (O(n^2)) – fine for small n
    dip = 0.0
    for i in range(n):
        for j in range(i, n):
            # candidate unimodal CDF: constant slope between i and j
            # Here we use a simple approximation: the CDF of a uniform distribution on [x_i, x_j]
            if sorted_data[j] == sorted_data[i]:
                continue
            F_unif = np.zeros_like(sorted_data)
            F_unif[sorted_data < sorted_data[i]] = 0.0
            F_unif[(sorted_data >= sorted_data[i]) & (sorted_data <= sorted_data[j])] = \
                (sorted_data[(sorted_data >= sorted_data[i]) & (sorted_data <= sorted_data[j])] - sorted_data[i]) / (sorted_data[j] - sorted_data[i])
            F_unif[sorted_data > sorted_data[j]] = 1.0
            dip = max(dip, np.max(np.abs(F_n - F_unif)))
    return dip

# Toy model with a single LoRA layer
class LoRALayer(nn.Module):
    def __init__(self, in_features, out_features, rank=4):
        super().__init__()
        self.weight = nn.Parameter(torch.randn(out_features, in_features) * 0.01)
        self.lora_A = nn.Parameter(torch.randn(rank, in_features) * 0.01)
        self.lora_B = nn.Parameter(torch.randn(out_features, rank) * 0.01)
        self.rank = rank

    def forward(self, x):
        # x: [batch, in_features]
        lora_update = (self.lora_B @ self.lora_A)  # [out_features, in_features]
        return F.linear(x, self.weight + lora_update)

def train_lambda(lambda_l2, steps=200, batch_size=32):
    model = LoRALayer(in_features=20, out_features=5)
    optimizer = torch.optim.SGD(model.parameters(), lr=0.1)
    # Dominant low‑rank direction (initially from lora_B @ lora_A)
    with torch.no_grad():
        dominant_dir = (model.lora_B @ model.lora_A).flatten()
        dominant_dir = dominant_dir / dominant_dir.norm()
    scores = []
    for step in range(steps):
        # Random synthetic data
        x = torch.randn(batch_size, 20)
        y = torch.randint(0, 5, (batch_size,))
        optimizer.zero_grad()
        logits = model(x)
        loss = F.cross_entropy(logits, y)
        # Frustration‑like L2 penalty on LoRA update vector
        lora_update = (model.lora_B @ model.lora_A)
        loss += lambda_l2 * lora_update.norm() ** 2
        loss.backward()
        optimizer.step()
        # Record alignment score for this step
        with torch.no_grad():
            update_flat = lora_update.flatten()
            align = (update_flat @ dominant_dir) / (update_flat.norm() * dominant_dir.norm() + 1e-8)
            scores.append(align.item())
    return np.array(scores)

lambdas = [0.0, 0.01, 0.1, 1.0]
results = {}
for lam in lambdas:
    scores = train_lambda(lam, steps=500)
    dip = hartigans_dip(scores)
    # Fit 1‑ and 2‑component GMM
    gmm1 = GaussianMixture(n_components=1, random_state=0).fit(scores.reshape(-1,1))
    gmm2 = GaussianMixture(n_components=2, random_state=0).fit(scores.reshape(-1,1))
    bic1 = gmm1.bic(scores.reshape(-1,1))
    bic2 = gmm2.bic(scores.reshape(-1,1))
    delta_bic = bic2 - bic1  # negative => 2‑component better
    results[lam] = {"dip_p": dip, "delta_bic": delta_bic, "scores": scores}
    print(f"λ={lam:.2f}: dip={dip:.4f}, ΔBIC={delta_bic:.2f}")

# Interpretation: increasing λ should raise dip (more multimodality) and make ΔBIC negative
