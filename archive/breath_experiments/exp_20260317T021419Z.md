We will test whether perfectly reversible (PIF‑style) LoRA perturbations preserve linear‑probe accuracy, and whether applying their exact inverse restores performance. We train a small vision classifier, freeze its backbone, inject a low‑rank LoRA delta, measure probe accuracy, then apply the negative of that delta and re‑measure. As a control, we compare against an equivalent magnitude irreversible drift obtained by a few SGD steps on the LoRA parameters.

```python
import torch, torch.nn as nn, torchvision.transforms as T, torchvision.datasets as D
from torch.utils.data import DataLoader

# Simple CNN backbone
class Backbone(nn.Module):
    def __init__(self):
        super().__init__()
        self.conv = nn.Sequential(
            nn.Conv2d(3, 16, 3, padding=1), nn.ReLU(),
            nn.Conv2d(16, 32, 3, padding=1), nn.ReLU(),
            nn.AdaptiveAvgPool2d(1)
        )
        self.fc = nn.Linear(32, 10)

    def forward(self, x):
        return self.fc(self.conv(x).view(x.size(0), -1))

# LoRA wrapper (low-rank additive)
class LoRALayer(nn.Module):
    def __init__(self, layer, rank=4):
        super().__init__()
        self.layer = layer
        self.rank = rank
        self.lora_A = nn.Parameter(torch.randn(rank, layer.in_features) * 0.01)
        self.lora_B = nn.Parameter(torch.randn(layer.out_features, rank) * 0.01)

    def forward(self, x):
        return self.layer(x) + (x @ self.lora_A.T @ self.lora_B.T)

# Prepare data
transform = T.Compose([T.ToTensor(), T.Normalize((0.5,)*3, (0.5,)*3)])
train_set = D.CIFAR10(root='./data', train=True, download=True, transform=transform)
test_set  = D.CIFAR10(root='./data', train=False, download=True, transform=transform)
train_loader = DataLoader(train_set, batch_size=128, shuffle=True)
test_loader  = DataLoader(test_set, batch_size=256)

# Initialize backbone and attach LoRA to final fc
model = Backbone()
model.fc = LoRALayer(model.fc, rank=4)
optimizer = torch.optim.SGD(model.parameters(), lr=0.1, momentum=0.9)
criterion = nn.CrossEntropyLoss()

# Train for a few epochs (just to get a reasonable classifier)
model.train()
for epoch in range(3):
    for x, y in train_loader:
        optimizer.zero_grad()
        loss = criterion(model(x), y)
        loss.backward()
        optimizer.step()

# Freeze backbone (except LoRA params) and train a linear probe on frozen features
model.eval()
with torch.no_grad():
    feats = []
    labels = []
    for x, y in test_loader:
        f = model.conv(x).view(x.size(0), -1)
        feats.append(f)
        labels.append(y)
    feats = torch.cat(feats)
    labels = torch.cat(labels)

probe = nn.Linear(feats.shape[1], 10)
probe_opt = torch.optim.SGD(probe.parameters(), lr=0.1)
for _ in range(20):
    probe_opt.zero_grad()
    logits = probe(feats)
    loss = criterion(logits, labels)
    loss.backward()
    probe_opt.step()

# Helper to compute probe accuracy
def probe_acc():
    probe.eval()
    with torch.no_grad():
        logits = probe(feats)
        pred = logits.argmax(dim=1)
        return (pred == labels).float().mean().item()

base_acc = probe_acc()
print(f"Base probe accuracy: {base_acc:.4f}")

# --- Reversible perturbation test ---
# Sample a random LoRA delta (same shape as lora_B) and apply it
delta = torch.randn_like(model.fc.lora_B) * 0.02  # small magnitude
with torch.no_grad():
    model.fc.lora_B += delta          # forward perturbation
acc_after_pert = probe_acc()
print(f"Accuracy after +δ: {acc_after_pert:.4f}")

# Apply exact inverse (negative delta) to undo
with torch.no_grad():
    model.fc.lora_B -= delta          # inverse perturbation
acc_after_inverse = probe_acc()
print(f"Accuracy after -δ (inverse): {acc_after_inverse:.4f}")

# --- Irreversible drift control ---
# Perform a few SGD steps on LoRA parameters using random labels (to cause drift)
model.train()
dummy_opt = torch.optim.SGD([model.fc.lora_B], lr=0.01)
for _ in range(5):
    dummy_opt.zero_grad()
    # random target to ensure gradient is not zero
    rand_target = torch.randint_like(labels, 0, 10)
    logits = model(torch.cat([x for x, _ in test_loader], dim=0)[:labels.size(0)])
    loss = criterion(logits, rand_target)
    loss.backward()
    dummy_opt.step()
model.eval()
acc_after_drift = probe_acc()
print(f"Accuracy after irreversible drift steps: {acc_after_drift:.4f}")
```

## 2. Extension #1 – Vision Transformers + *multiple* linear probes
