We will simulate a Kuramoto oscillator network, compute the histogram peak of the order‑parameter R, and construct a Goal‑Drift Index (GDI) from peak location, bimodality coefficient, distribution entropy, and short‑time autocorrelation. By applying a uniform phase shift to all oscillators (a pure translation) we expect the peak to move while the GDI remains unchanged; any significant GDI change would indicate a genuine reorganization of the synchrony landscape.

```python
import numpy as np

def kuramoto_step(theta, omega, K, dt=0.01):
    N = len(theta)
    pair_diff = theta[:, None] - theta[None, :]  # NxN
    dtheta = omega + (K/N) * np.sum(np.sin(pair_diff), axis=1)
    return theta + dtheta * dt

def simulate(N=100, K=0.5, T=200, dt=0.01, shift=0.0):
    theta = np.random.uniform(0, 2*np.pi, N)
    omega = np.random.normal(0, 0.5, N)
    Rs = []
    for _ in range(int(T/dt)):
        theta = kuramoto_step(theta, omega, K, dt)
        R = np.abs(np.sum(np.exp(1j*theta))) / N
        Rs.append(R)
        # apply uniform shift after each step to test translation invariance
        theta = (theta + shift) % (2*np.pi)
    return np.array(Rs)

def histogram_peak(Rs, bins=50):
    hist, edges = np.histogram(Rs, bins=bins, density=True)
    peak_idx = np.argmax(hist)
    return (edges[peak_idx] + edges[peak_idx+1])/2

def bimodality_coefficient(Rs):
    from scipy.stats import skew, kurtosis
    b = (skew(Rs)**2 + 1) / (kurtosis(Rs) + 3*((len(Rs)-1)**2)/((len(Rs)-2)*(len(Rs)-3)))
    return b

def entropy_dist(Rs, bins=50):
    hist, _ = np.histogram(Rs, bins=bins, density=True)
    hist = hist[hist>0]
    return -np.sum(hist * np.log(hist))

def autocorr_lag1(Rs):
    return np.corrcoef(Rs[:-1], Rs[1:])[0,1]

def compute_GDI(Rs):
    peak = histogram_peak(Rs)
    bc = bimodality_coefficient(Rs)
    ent = entropy_dist(Rs)
    ac = autocorr_lag1(Rs)
    # normalize each component to zero mean, unit variance across a baseline set
    # here we simply return the raw vector; in practice you would z‑score
    return np.array([peak, bc, ent, ac])

# Baseline simulation (no shift)
Rs_base = simulate(shift=0.0)
gdi_base = compute_GDI(Rs_base)
peak_base = histogram_peak(Rs_base)

# Translation simulation (add constant phase shift each step)
Rs_shift = simulate(shift=0.2)   # 0.2 rad per integration step
gdi_shift = compute_GDI(Rs_shift)
peak_shift = histogram_peak(Rs_shift)

print("Peak shift:", peak_shift - peak_base)
print("GDI difference (norm):", np.linalg.norm(gdi_shift - gdi_base))
```
