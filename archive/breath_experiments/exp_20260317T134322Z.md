## Proposal
PROBE
We will train a simple neural network on a synthetic dataset and compute the mutual information between successive weight updates and the instantaneous loss using a neural mutual information estimator (MINE). If low‑MSE steps indeed carry higher mutual information, we should observe a positive correlation between low loss values and estimated MI across training steps.

```python
import torch
import torch.nn as nn
import torch.optim as optim
import numpy as np

# Simple MINE estimator
class MINE(nn.Module):
    def __init__(self, x_dim, y_dim, hidden_size=128):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(x_dim + y_dim, hidden_size),
            nn.ReLU(),
            nn.Linear(hidden_size, 1)
        )
    def forward(self, x, y):
        xy = torch.cat([x, y], dim=-1)
        return self.net(xy)
    def mutual_information(self, x, y):
        # shuffled samples for marginal
        y_shuffled = y[torch.randperm(y.size(0))]
        joint = self.forward(x, y)
        marginal = self.forward(x, y_shuffled)
        return (joint.mean() - torch.log(torch.exp(marginal).mean())).detach()

# Toy regression problem
torch.manual_seed(0)
X = torch.randn(200, 10)
true_w = torch.randn(10, 1)
y = X @ true_w + 0.1 * torch.randn(200, 1)

model = nn.Linear(10, 1)
optimizer = optim.SGD(model.parameters(), lr=0.1)
criterion = nn.MSELoss()

mine = MINE(x_dim=10, y_dim=1)  # weight update dim, loss dim
mine_opt = optim.Adam(mine.parameters(), lr=1e-3)

losses = []
weight_updates = []
mis = []

for epoch in range(300):
    optimizer.zero_grad()
    pred = model(X)
    loss = criterion(pred, y)
    loss.backward()
    
    # store current weight update (gradient * lr) as a vector
    with torch.no_grad():
        update = -optimizer.param_groups[0]['lr'] * torch.cat(
            [p.grad.view(-1) for p in model.parameters() if p.grad is not None]
        )
        weight_updates.append(update.clone())
        losses.append(loss.item())
    
    optimizer.step()
    
    # Train MINE for a few steps
    for _ in range(5):
        mine_opt.zero_grad()
        x_batch = torch.stack(weight_updates[-min(32, len(weight_updates)):])
        y_batch = torch.tensor(losses[-min(32, len(losses)):]).unsqueeze(-1)
        joint = mine(x_batch, y_batch)
        y_shuffled = y_batch[torch.randperm(y_batch.size(0))]
        marginal = mine(x_batch, y_shuffled)
        loss_mine = -(joint.mean() - torch.log(torch.exp(marginal).mean()))
        loss_mine.backward()
        mine_opt.step()
    
    # estimate MI
    with torch.no_grad():
        x_batch = torch.stack(weight_updates[-min(32, len(weight_updates)):])
        y_batch = torch.tensor(losses[-min(32, len(losses)):]).unsqueeze(-1)
        mi_est = mine.mutual_information(x_batch, y_batch)
        mis.append(mi_est.item())

# Compute correlation between low loss and MI
losses = np.array(losses)
mis = np.array(mis)
# low loss -> higher MI => negative correlation
corr = np.corrcoef(losses, mis)[0,1]
print(f"Pearson correlation between loss and estimated MI: {corr:.3f}")
```
