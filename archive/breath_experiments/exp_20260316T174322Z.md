import numpy as np
import torch
import torch.nn as nn
import torch.optim as optim

# ------------------------------------------------------------
# Settings
# ------------------------------------------------------------
dim = 2
n_samples = 10_000          # training set size
n_test    = 2_000           # test set size
noise_scale = 0.5           # std of the scalar‑noise distribution (mean=1)
sigma_iso   = 0.5           # std of each component of isotropic Gaussian noise
torch.manual_seed(0)
np.random.seed(0)

# True gradient (fixed direction for simplicity)
true_grad = np.array([1.0, 0.0])          # unit vector along x‑axis
true_grad_t = torch.tensor(true_grad, dtype=torch.float32)

# ------------------------------------------------------------
# Data generation
# ------------------------------------------------------------
def generate_data(noise_type: str, size: int):
    """
    Returns:
        X          : torch.Tensor, shape (size, 2) – noisy gradient samples
        Y_phase    : torch.Tensor, shape (size, 1) – signed angle (rad) between noisy and true
        Y_scale    : torch.Tensor, shape (size, 1) – optimal scalar factor that makes
                     ||scale * noisy|| = ||true|| (i.e. scale = ||true||/||noisy||)
    """
    X_list = []
    phase_list = []
    scale_list = []

    for _ in range(size):
        if noise_type == 'scalar':
            # scalar noise: real scaling factor s ~ N(1, noise_scale^2)
            s = np.random.normal(loc=1.0, scale=noise_scale)
            noisy = s * true_grad
        else:  # isotropic Gaussian
            noisy = true_grad + np.random.normal(loc=0.0, scale=sigma_iso, size=dim)

        noisy_t = torch.tensor(noisy, dtype=torch.float32)

        # ---- phase (signed angle) -------------------------------------------------
        # Using atan2 gives a signed angle in (-π, π] and avoids the acos+sign hack.
        # Angle from true gradient (pointing along +x) to noisy vector:
        phase = np.arctan2(noisy[1], noisy[0])   # true gradient has angle 0
        # ---- optimal scalar scale -------------------------------------------------
        norm_noisy = np.linalg.norm(noisy)
        scale = 1.0 / norm_noisy if norm_noisy > 0 else 1.0   # ||true|| = 1

        X_list.append(noisy_t)
        phase_list.append(phase)
        scale_list.append(scale)

    X = torch.stack(X_list)                     # (size, 2)
    Y_phase = torch.tensor(phase_list, dtype=torch.float32).unsqueeze(1)  # (size,1)
    Y_scale = torch.tensor(scale_list, dtype=torch.float32).unsqueeze(1)  # (size,1)
    return X, Y_phase, Y_scale


# ------------------------------------------------------------
# Simple MLP predictor
# ------------------------------------------------------------
class Predictor(nn.Module):
    def __init__(self, out_dim):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(dim, 64),
            nn.ReLU(),
            nn.Linear(64, 64),
            nn.ReLU(),
            nn.Linear(64, out_dim)
        )
    def forward(self, x):
        return self.net(x)


def train_model(X, Y, epochs=200, lr=1e-3):
    model = Predictor(Y.shape[1])
    criterion = nn.MSELoss()
    optimizer = optim.Adam(model.parameters(), lr=lr)
    for epoch in range(epochs):
        optimizer.zero_grad()
        pred = model(X)
        loss = criterion(pred, Y)
        loss.backward()
        optimizer.step()
    return model


# ------------------------------------------------------------
# Experiment loop
# ------------------------------------------------------------
def run_experiment(noise_type: str):
    print(f'\n=== Noise type: {noise_type} ===')
    # ----- training data -----
    X_train, Y_phase_train, Y_scale_train = generate_data(noise_type, n_samples)

    # ----- test data -----
    X_test, Y_phase_test, Y_scale_test = generate_data(noise_type, n_test)

    # ----- train predictors -----
    model_phase = train_model(X_train, Y_phase_train)
    model_scale = train_model(X_train, Y_scale_train)

    # ----- evaluate on test set -----
    with torch.no_grad():
        # predictions
        phase_pred = model_phase(X_test).squeeze()          # (n_test,)
        scale_pred = model_scale(X_test).squeeze()          # (n_test,)

        # ----- denoise using predicted phase (rotate back) -----
        # Treat each noisy vector as a complex number z = x + iy
        z = X_test[:, 0] + 1j * X_test[:, 1]                # (n_test,)
        z_rot = z * torch.exp(-1j * phase_pred)            # rotate by -phase_pred
        denoised_grad = torch.stack([z_rot.real, z_rot.imag], dim=1)  # (n_test,2)

        # ----- denoise using predicted scale (simple scaling) -----
        denoised_grad_scale = X_test * scale_pred.unsqueeze(1)

        # ----- alignment (cosine similarity) with the true gradient -----
        true_batch = true_grad_t.unsqueeze(0).expand_as(denoised_grad)  # (n_test,2)
        align_phase = torch.nn.functional.cosine_similarity(denoised_grad,
                                                            true_batch,
                                                            dim=1).mean().item()
        align_scale = torch.nn.functional.cosine_similarity(denoised_grad_scale,
                                                            true_batch,
                                                            dim=1).mean().item()

        # ----- prediction errors (MSE) -----
        mse_phase = torch.nn.functional.mse_loss(phase_pred, Y_phase_test.squeeze()).item()
        mse_scale = torch.nn.functional.mse_loss(scale_pred, Y_scale_test.squeeze()).item()

    # ----- report -----
    print(f'  Phase predictor MSE: {mse_phase:.5f}')
    print(f'  Scale predictor MSE: {mse_scale:.5f}')
    print(f'  Mean alignment after phase correction: {align_phase:.5f}')
    print(f'  Mean alignment after scale correction: {align_scale:.5f}')


# ------------------------------------------------------------
# Run both experiments
# ------------------------------------------------------------
if __name__ == '__main__':
    run_experiment('scalar')
    run_experiment('isotropic')
