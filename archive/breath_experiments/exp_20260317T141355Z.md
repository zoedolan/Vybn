We expect the aligned case to yield a significantly higher MI, demonstrating that probe‑signal alignment is essential for detecting hidden dependencies.

import numpy as np

def mi_estimator(x, y, k=5):
    """Simple k‑NN mutual information estimator (Kraskov et al., 2004)."""
    # x: (n_samples, n_x), y: (n_samples, n_y)
    n = x.shape[0]
    # concatenate for joint space
    xy = np.concatenate([x, y], axis=1)
    # compute distances to k-th neighbor in joint and marginal spaces
    from sklearn.neighbors import NearestNeighbors
    nbrs_xy = NearestNeighbors metric='chebyshev').fit(xy)
    dist_xy, _ = nbrs_xy.kneighbors(xy, n_neighbors=k+1)
    eps_xy = dist_xy[:, -1]  # distance to k-th neighbor (exclude self)

    nbrs_x = NearestNeighbors(metric='chebyshev').fit(x)
    nbrs_y = NearestNeighbors(metric='chebyshev').fit(y)
    # count points within eps_xy-1e-15 in marginals
    nx = nbrs_x.radius_neighbors(x, radius=eps_xy-1e-15, return_distance=False)
    ny = nbrs_y.radius_neighbors(y, radius=eps_xy-1e-15, return_distance=False)
    nx = np.array([len(nb)-1 for nb in nx])  # exclude self
    ny = np.array([len(nb)-1 for nb in ny])
    # digamma approximation
    from scipy.special import digamma
    mi = digamma(k) + digamma(n) - np.mean(digamma(nx+1) + digamma(ny+1))
    return mi

# Simulate data: weight updates influence loss with a known nonlinear component
np.random.seed(42)
n_samples = 500
# latent variable driving both
z = np.random.randn(n_samples, 1)
# weight updates: linear + noise
wu = 0.8 * z + 0.2 * np.random.randn(n_samples, 2)
# loss: nonlinear function of z plus noise
loss = np.sin(3 * z).ravel() + 0.1 * np.random.randn(n_samples)

# Aligned probe: flatten weight updates to (n_samples, 2)
wu_aligned = wu  # shape (500,2)
mi_aligned = mi_estimator(wu_aligned, loss.reshape(-1,1))

# Mis‑aligned probe: incorrectly transpose to (2, n_samples) then flatten incorrectly
wu_mis = wu.T  # shape (2,500)
# Pretend we feed this as (n_samples,2) by reshaping incorrectly
wu_mis_feat = wu_mis.reshape(n_samples, -1)  # still (500,2) but columns swapped
# Actually the information is scrambled; we further rotate to destroy alignment
wu_mis_feat = np.roll(wu_mis_feat, shift=1, axis=1)  # shift columns
mi_mis = mi_estimator(wu_mis_feat, loss.reshape(-1,1))

print(f"Mutual information (aligned): {mi_aligned:.4f}")
print(f"Mutual information (mis‑aligned): {mi_mis:.4f}")
