We will run a few iterations and record whether the mean R trends upward, demonstrating that GDI can serve as a reward for structural refinements.

```python
import numpy as np

# Parameters
N = 100               # number of oscillators
T = 200               # simulation steps
dt = 0.01             # time step
sigma = 0.05          # noise std
K_base = 0.5          # baseline coupling strength

# Natural frequencies (quenched disorder)
np.random.seed(0)
omega = np.random.normal(0, 1, N)

# Initial phases
theta = np.random.uniform(0, 2*np.pi, N)

# Coupling matrix: start with all-to-all uniform
K = np.full((N, N), K_base / N)
np.fill_diagonal(K, 0.0)

def kuramoto_step(theta, K, omega, dt, sigma):
    """Return dθ for the Kuramoto model with shape (N,)"""
    # Pairwise phase differences
    delta = theta[:, None] - theta[None, :]  # (N, N)
    coupling = K * np.sin(delta)            # (N, N)
    dtheta = omega + coupling.sum(axis=1)   # (N,)
    # Add Euler-Maruyama noise
    dtheta += sigma * np.random.randn(N) * np.sqrt(dt)
    return dtheta

def simulate(theta0, K, steps=T, dt=dt, sigma=sigma):
    """Run simulation and return time series of order parameter R."""
    theta = theta0.copy()
    R_series = []
    for _ in range(steps):
        dtheta = kuramoto_step(theta, K, omega, dt, sigma)
        theta += dtheta * dt
        # Order parameter R = |⟨e^{iθ}⟩|
        R = np.abs(np.sum(np.exp(1j * theta))) / N
        R_series.append(R)
    return np.array(R_series)

def histogram(R_series, bins=20):
    hist, _ = np.histogram(R_series, bins=bins, density=True)
    return hist

# Initial simulation
R0 = simulate(theta, K)
hist0 = histogram(R0)
mean_R0 = R0.mean()

# ICPO‑style refinement loop
n_iters = 5
accept_history = []
mean_R_history = [mean_R0]

for it in range(n_iters):
    # Propose a topological tweak: add a small random perturbation to a subset of couplings
    K_prop = K.copy()
    idx = np.random.choice(N, size=5, replace=False)  # pick 5 nodes
    for i in idx:
        for j in idx:
            if i != j:
                K_prop[i, j] += np.random.normal(0, 0.05)  # small random change
                K_prop[j, i] = K_prop[i, j]                # keep symmetry
    # Simulate with proposed coupling
    R_prop = simulate(theta, K_prop)
    hist_prop = histogram(R_prop)
    # Goal‑Drift Index: L1 distance between histograms (larger = more drift)
    GDI = np.sum(np.abs(hist_prop - hist0))
    # Accept if mean R increases (higher coherence) – using GDI as feedback signal
    if R_prop.mean() > mean_R_history[-1]:
        K = K_prop
        accept_history.append(True)
    else:
        accept_history.append(False)
    mean_R_history.append(R_prop.mean())
    print(f"Iter {it+1}: GDI={GDI:.4f}, Δmean R={mean_R_history[-1]-mean_R_history[-2]:.4f}, Accept={accept_history[-1]}")

print("\nAcceptance record:", accept_history)
print("Mean R over iterations:", mean_R_history)
```
