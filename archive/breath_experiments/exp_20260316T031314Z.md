def dirichlet_sequence(alpha, length=5000, seed=0):
    rng = np.random.default_rng(seed)

def covariance_trace(samples):
    # samples shape (T, D)
    cov = np.cov(samples, rowvar=False)  # D x D
    return np.trace(cov)
