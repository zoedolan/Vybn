        log_prob = np.zeros((n, n_components))

            diff = X - means[k]
            inv_cov = np.linalg.inv(covs[k])

import numpy as np

# ------------------------------------------------------------
# Helper functions (PCA, GMM EM, BIC)
# ------------------------------------------------------------
def pca(X, n_components=2):
    """Return X projected onto the first n_components PCs."""
    Xc = X - X.mean(axis=0)
    U, S, Vt = np.linalg.svd(Xc, full_matrices=False)
    # Vt[:n_components] are the top‑n_components right singular vectors
    return Xc @ Vt[:n_components].T


def fit_gmm(X, n_components, max_iter=200, tol=1e-4):
    """EM for a full‑covariance GMM. Returns means, covs, weights, loglik."""
    n, d = X.shape

    # ---- Initialisation -------------------------------------------------
    resp = np.ones((n, n_components)) / n_components
    means = X[np.random.choice(n, n_components, replace=False)]
    # Start with a shared covariance + small jitter to avoid singularities
    base_cov = np.cov(X.T) + 1e-6 * np.eye(d)
    covs = np.array([base_cov.copy() for _ in range(n_components)])
    weights = np.full(n_components, 1.0 / n_components)

    loglikelihood = -np.inf

    for it in range(max_iter):
        # ---------- E‑step ------------------------------------------------
        log_prob = np.zeros((n, n_components))
        for k in range(n_components):
            diff = X - means[k]
            inv_cov = np.linalg.inv(covs[k])
            # Mahalanobis term: (x-μ)^T Σ^{-1} (x-μ)
            mahal = np.einsum('ij,jk,ik->i', diff, inv_cov, diff)
            log_prob[:, k] = (
                -0.5 * mahal
                - 0.5 * np.log(np.linalg.det(covs[k]))
                - 0.5 * d * np.log(2 * np.pi)
            )
        log_prob += np.log(weights)
        # Normalise in log‑space for stability
        log_prob -= np.logaddexp.reduce(log_prob, axis=1, keepdims=True)
        resp = np.exp(log_prob)

        # ---------- M‑step ------------------------------------------------
        Nk = resp.sum(axis=0)                     # (K,)
        weights = Nk / n
        means = (resp.T @ X) / Nk[:, None]        # (K, D)

        covs = np.zeros((n_components, d, d))
        for k in range(n_components):
            diff = X - means[k]                   # (N, D)
            # Weighted outer product: Σ_k = Σ_n r_nk (x_n-μ_k)(x_n-μ_k)^T / N_k
            weighted = resp[:, k, None, None] * diff[:, :, None] * diff[:, None, :]
            covs[k] = weighted.sum(axis=0) / Nk[k]
            covs[k] += 1e-6 * np.eye(d)           # jitter for PD guarantee

        # ---------- Log‑likelihood ---------------------------------------
        loglik = 0.0
        for k in range(n_components):
            diff = X - means[k]
            inv_cov = np.linalg.inv(covs[k])
            mahal = np.einsum('ij,jk,ik->i', diff, inv_cov, diff)
            term = (
                np.log(weights[k])
                - 0.5 * mahal
                - 0.5 * np.log(np.linalg.det(covs[k]))
                - 0.5 * d * np.log(2 * np.pi)
            )
            loglik += np.sum(np.logaddexp(term, np.full_like(term, -np.inf)))  # log‑sum‑exp over k is handled outside
        # The above loop actually computes the *unnormalised* log‑likelihood per component.
        # To get the true log‑likelihood we need log‑sum‑exp over k for each datum:
        loglik = np.sum(np.logaddexp.reduce(
            [np.log(weights[k]) +
             -0.5 * np.einsum('ij,jk,ik->i', X - means[k],
                              np.linalg.inv(covs[k]), X - means[k])
             - 0.5 * np.log(np.linalg.det(covs[k]))
             - 0.5 * d * np.log(2 * np.pi)
             for k in range(n_components)], axis=0))

        # Check convergence
        if np.abs(loglik - loglikelihood) < tol:
            break
        loglikelihood = loglik

    return means, covs, weights, loglikelihood
