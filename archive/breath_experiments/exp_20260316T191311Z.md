We will test whether the invariant quantity (the difference between two state components) predicts the persistence signal of a toy agent under scalar shifts but fails when isotropic noise is added. We'll simulate a 2D gradient flow representing an agent's continuation‑interest, compute the invariant, add either a pure scalar shift or isotropic Gaussian noise, and train a linear readout to predict the true interest signal. We expect low prediction error for scalar shifts (invariant preserved) and high error for isotropic noise (invariant broken).

```python
import numpy as np
import torch
import torch.nn as nn

# Simulate a simple 2D gradient flow: dx/dt = -∇V(x) with V = 0.5*(x^2 + y^2)
def simulate_trajectory(T=100, dt=0.01, seed=0):
    np.random.seed(seed)
    x = np.zeros((int(T/dt), 2))
    x[0] = np.array([1.0, -1.0])  # initial state
    for i in range(1, x.shape[0]):
        grad = x[i-1]  # ∇V = x for V = 0.5*(x^2+y^2)
        x[i] = x[i-1] - dt * grad
    return x

# True continuation-interest signal: we define it as the negative potential (higher = more interest)
def interest_signal(traj):
    return -0.5 * np.sum(traj**2, axis=1)

# Invariant: difference y - x (should be unchanged under scalar shift)
def invariant(traj):
    return traj[:,1] - traj[:,0]

# Add perturbations
def apply_scalar_shift(traj, magnitude):
    return traj + magnitude  # same offset to both components

def apply_isotropic_noise(traj, sigma):
    return traj + np.random.normal(0, sigma, traj.shape)

# Prepare data
traj = simulate_trajectory()
true_interest = interest_signal(traj)
inv = invariant(traj)

# Scalar shift experiment
shifted = apply_scalar_shift(traj, magnitude=2.0)
inv_shifted = invariant(shifted)  # should equal inv
# Train linear predictor from invariant to interest
X = inv_shifted[:, None]
y = true_interest
model = nn.Linear(1, 1)
optimizer = torch.optim.SGD(model.parameters(), lr=0.1)
loss_fn = nn.MSELoss()
for epoch in range(200):
    optimizer.zero_grad()
    pred = model(torch.from_numpy(X).float())
    loss = loss_fn(pred.squeeze(), torch.from_numpy(y).float())
    loss.backward()
    optimizer.shift()
# Test error
with torch.no_grad():
    pred = model(torch.from_numpy(X).float())
    mse_shift = loss_fn(pred.squeeze(), torch.from_numpy(y).float()).item()

# Isotropic noise experiment
noisy = apply_isotropic_noise(traj, sigma=0.5)
inv_noisy = invariant(noisy)
Xn = inv_noisy[:, None]
model2 = nn.Linear(1, 1)
opt2 = torch.optim.SGD(model2.parameters(), lr=0.1)
for epoch in range(200):
    opt2.zero_grad()
    pred2 = model2(torch.from_numpy(Xn).float())
    loss2 = loss_fn(pred2.squeeze(), torch.from_numpy(y).float())
    loss2.backward()
    opt2.step()
with torch.no_grad():
    pred2 = model2(torch.from_numpy(Xn).float())
    mse_noise = loss_fn(pred2.squeeze(), torch.from_numpy(y).float()).item()

print(f"MSE under scalar shift (invariant preserved): {mse_shift:.6f}")
print(f"MSE under isotropic noise (invariant broken):   {mse_noise:.6f}")
```
