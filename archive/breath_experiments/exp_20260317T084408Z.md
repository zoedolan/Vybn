Test whether LoRA update vectors (Δθ) form two distinct clusters that correspond to high vs. low frustration penalty spikes, indicating an internal regime shift driven by the reward signal.
We will collect Δθ at each training step, compute the frustration penalty value, label steps with penalty above/below a threshold, fit a Gaussian Mixture Model (GMM) with two components to the Δθ, and evaluate cluster separation using silhouette score and the Hartigan’s dip test for unimodality.

```python
import numpy as np
import torch
from sklearn.mixture import GaussianMixture
from sklearn.metrics import silhouette_score
from scipy.stats import diptest

# Simulated data: replace with actual LoRA updates and frustration values
num_steps = 2000
update_dim = 128
# Generate bimodal updates: high frustration -> large norm, low frustration -> small norm
high_mask = np.random.rand(num_steps) > 0.7
updates = np.random.randn(num_steps, update_dim) * 0.1
updates[high_mask] += np.random.randn(num_steps[high_mask].sum(), update_dim) * 1.0
frustration = np.random.rand(num_steps)
frustration[high_mask] += 0.5  # spike frustration for high updates

# Label by frustration threshold
threshold = np.percentile(frustration, 70)
labels = (frustration > threshold).astype(int)

# Fit GMM with two components on updates
gmm = GaussianMixture(n_components=2, covariance_type='full', random_state=0)
gmm.fit(updates)
gmm_labels = gmm.predict(updates)

# Silhouette score (higher -> better separation)
sil = silhouette_score(updates, gmm_labels)
print(f'Silhouette score: {sil:.3f}')

# Hartigan's dip test on projected 1D direction (first PCA component)
from sklearn.decomposition import PCA
pca = PCA(n_components=1)
proj = pca.fit_transform(updates).flatten()
dip_stat, p_val = diptest(proj)
print(f'Hartigan dip statistic: {dip_stat:.4f}, p-value: {p_val:.4f}')
```
